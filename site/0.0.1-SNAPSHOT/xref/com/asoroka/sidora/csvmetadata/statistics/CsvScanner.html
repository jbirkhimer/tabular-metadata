21  /**
22   * The heart of statistical workflow. Handed a {@link CSVParser}, this class will scan through it and supply the
23   * values of fields to a "row" of {@link DataTypeHeuristic} strategies cloned from the configured choice.
24   * 
25   * @author ajs6f
26   */
27  public class CsvScanner extends AbstractIterator<CSVRecord> {
28  
29      private final PeekingIterator<CSVRecord> internalScanner;
30  
31      private final Iterator<DataTypeHeuristic<?>> strategies;
32  
33      private final List<DataTypeHeuristic<?>> rowOfStrategies;
34  
35      private static final Logger log = getLogger(CsvScanner.class);
36  
37      public CsvScanner(final CSVParser parser, final DataTypeHeuristic<?> strategy) {
38          super();
39          this.internalScanner = peekingIterator(parser.iterator());
40          final int numColumns = internalScanner.peek().size();
41  
42          log.debug("Found {} columns in our CSV.", numColumns);
43          // a "row" of strategy instances of the same length as the rows in our CSV
44          this.rowOfStrategies = newArrayListWithCapacity(numColumns);
45          for (int i = 0; i < numColumns; i++) {
46              rowOfStrategies.add(strategy.clone());
47          }
48          // this.strategies will cycle endlessly around our row
49          this.strategies = cycle(rowOfStrategies);
50      }
51  
52      @Override
53      protected CSVRecord computeNext() {
54          if (internalScanner.hasNext()) {
55              final CSVRecord nextRecord = internalScanner.next();
56              for (final String value : nextRecord) {
57                  strategies.next().addValue(value);
58              }
59              return nextRecord;
60          }
61          return endOfData();
62      }
63  
64      /**
65       * Scan rows in our CSV up to a limit.
66       * 
67       * @param limit The number of rows to scan, 0 for all rows.
68       */
69      public void scan(final int limit) {
70          if (limit == 0) {
71              while (hasNext()) {
72                  next();
73              }
74          } else {
75              advance(this, limit);
76          }
77      }
78  
79      /**
80       * Use this to recover and interrogate the strategy instances used in scanning.
81       * 
82       * @return the row of strategies used to determine the types of fields
83       */
84      public List<DataTypeHeuristic<?>> getStrategies() {
85          return rowOfStrategies;
86      }
