002package com.asoroka.sidora.csvmetadata;

004import static org.junit.Assert.assertEquals;
005
006import java.util.List;
007
008import org.junit.Test;
009import org.junit.runner.RunWith;
010import org.mockito.Mock;
011import org.mockito.runners.MockitoJUnitRunner;
012
013import com.asoroka.sidora.csvmetadata.datatype.DataType;
014import com.google.common.collect.Range;
015
016@RunWith(MockitoJUnitRunner.class)
017public class CsvMetadataTest {
018
019    @Mock
020    private List<String> mockHeaderNames;
021
022    @Mock
023    private List<DataType> mockTypes;
024
025    @Mock
026    private List<Range<?>> mockRanges;
027
028    @Test
029    public void testContainerAction() {
030        final CsvMetadata testMetadata = new CsvMetadata(mockHeaderNames, mockTypes, mockRanges);
031        assertEquals(mockHeaderNames, testMetadata.headerNames());
032        assertEquals(mockTypes, testMetadata.columnTypes());
033        assertEquals(mockRanges, testMetadata.minMaxes());
034    }
