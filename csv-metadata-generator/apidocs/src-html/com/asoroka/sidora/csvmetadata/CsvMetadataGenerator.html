002package com.asoroka.sidora.csvmetadata;
003
004import static com.google.common.base.Charsets.UTF_8;
005import static com.google.common.collect.Lists.newArrayList;
006import static com.google.common.collect.Lists.transform;
007import static java.util.Collections.emptyList;
008import static org.apache.commons.csv.CSVFormat.DEFAULT;
009import static org.apache.commons.csv.CSVParser.parse;
010
011import java.io.IOException;
012import java.net.URL;
013import java.nio.charset.Charset;
014import java.util.List;
015
016import javax.inject.Inject;
017
018import org.apache.commons.csv.CSVFormat;
019import org.apache.commons.csv.CSVParser;
020import org.apache.commons.csv.CSVRecord;
021
022import com.asoroka.sidora.csvmetadata.datatype.DataType;
023import com.asoroka.sidora.csvmetadata.formats.CsvFormat;
024import com.asoroka.sidora.csvmetadata.heuristics.DataTypeHeuristic;
025import com.asoroka.sidora.csvmetadata.heuristics.HeaderHeuristic;
026import com.asoroka.sidora.csvmetadata.heuristics.HeaderHeuristic.Default;
027import com.asoroka.sidora.csvmetadata.heuristics.StrictHeuristic;
028import com.asoroka.sidora.csvmetadata.statistics.CsvScanner;
029import com.google.common.base.Function;
030import com.google.common.collect.Range;
031
032public class CsvMetadataGenerator {
033
034    private static final Charset CHARACTER_ENCODING = UTF_8;
035
036    private CSVFormat format = DEFAULT;
037
038    /**
039     * Default value of {@code 0} indicates no limit. See {@link CsvScanner.scan(final int limit)}.
040     */
041    private Integer scanLimit = 0;
042
043    private DataTypeHeuristic<?> strategy = new StrictHeuristic();
044
045    private HeaderHeuristic headerStrategy = new Default();
046
047    public CsvMetadata getMetadata(final URL csvFile) throws IOException {
048
049        final List<String> headerNames;
050        try (final CSVParser headerParser = parse(csvFile, CHARACTER_ENCODING, format)) {
051            final CSVRecord firstLine = headerParser.iterator().next();
052            final boolean hasHeaders = headerStrategy.apply(firstLine);
053            format = hasHeaders ? format.withHeader() : format;
054            headerNames = hasHeaders ? newArrayList(firstLine) : emptyHeaders;
055        }
056
057        final List<DataTypeHeuristic<?>> strategies;
058        try (final CSVParser parser = parse(csvFile, CHARACTER_ENCODING, format)) {
059            final CsvScanner scanner = new CsvScanner(parser, strategy);
060            scanner.scan(scanLimit);
061            strategies = scanner.getStrategies();
062        }
063
064        final List<DataType> columnTypes = transform(strategies, extractType);
065        final List<Range<?>> minMaxes = transform(strategies, extractMinMax);
066
067        return new CsvMetadata(headerNames, columnTypes, minMaxes);
068    }
069
070    @TypeRenaming
071    private static interface Extractor<T> extends Function<DataTypeHeuristic<?>, T> {
072    }
073
074    private static final Extractor<DataType> extractType = new Extractor<DataType>() {
075
076        @Override
077        public DataType apply(final DataTypeHeuristic<?> strategy) {
078            return strategy.mostLikelyType();
079        }
080    };
081
082    private static final Extractor<Range<?>> extractMinMax = new Extractor<Range<?>>() {
083
084        @Override
085        public Range<?> apply(final DataTypeHeuristic<?> strategy) {
086            return strategy.getRange();
087        }
088    };
089
090    private List<String> emptyHeaders = emptyList();
091
092    public void setScanLimit(final Integer scanLimit) {
093        this.scanLimit = scanLimit;
094    }
095
096    @Inject
097    public void setHeaderStrategy(final HeaderHeuristic acceptor) {
098        this.headerStrategy = acceptor;
099    }
100
101    @Inject
102    public void setFormat(final CsvFormat format) {
103        this.format = format.get();
104    }
105
106    @Inject
107    public void setStrategy(final DataTypeHeuristic<?> strategy) {
108        this.strategy = strategy;
109    }
110}
