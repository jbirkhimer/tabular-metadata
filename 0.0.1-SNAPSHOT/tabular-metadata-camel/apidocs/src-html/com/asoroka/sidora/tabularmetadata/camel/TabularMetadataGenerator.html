002package com.asoroka.sidora.tabularmetadata.camel;
003
004import java.io.IOException;
005import java.net.URL;
006
007import org.apache.camel.Handler;
008import org.apache.camel.Header;
009
010import com.asoroka.sidora.tabularmetadata.TabularMetadata;
011
012/**
013 * A specialization of {@link com.asoroka.sidora.tabularmetadata.TabularMetadataGenerator} for use with Camel.
014 * 
015 * @author ajs6f
016 */
017public class TabularMetadataGenerator extends com.asoroka.sidora.tabularmetadata.TabularMetadataGenerator {
018
019    public static final String dataUrlHeaderName = "com.asoroka.sidora.tabularmetadata.camel.dataUrlHeaderName";
020
021    @Handler
022    public TabularMetadata getMetadata(@Header(dataUrlHeaderName) final String dataUrl) throws IOException {
023        return super.getMetadata(new URL(dataUrl));
024    }
025}
