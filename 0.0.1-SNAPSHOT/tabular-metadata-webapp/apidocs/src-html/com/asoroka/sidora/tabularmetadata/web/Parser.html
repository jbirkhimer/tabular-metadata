002package com.asoroka.sidora.tabularmetadata.web;
003
004import static org.slf4j.LoggerFactory.getLogger;
005
006import java.io.IOException;
007import java.net.URL;
008
009import javax.inject.Inject;
010import javax.ws.rs.GET;
011import javax.ws.rs.Path;
012import javax.ws.rs.Produces;
013import javax.ws.rs.QueryParam;
014
015import org.slf4j.Logger;
016
017import com.asoroka.sidora.tabularmetadata.TabularMetadataGenerator;
018
019@Path("/")
020public class Parser {
021
022    @Inject
023    private TabularMetadataGenerator generator;
024
025    private static final Logger log = getLogger(Parser.class);
026
027    @GET
028    @Path("/")
029    @Produces("text/xml")
030    public TabularMetadataPrecis get(@QueryParam("url") final URL url) throws IOException {
031        return new TabularMetadataPrecis(generator.getMetadata(url));
032    }
033}
