002package com.asoroka.sidora.tabularmetadata.web;
003
004import static com.google.common.base.Charsets.UTF_8;
005import static java.nio.file.Files.createTempFile;
006import static java.nio.file.Files.write;
007import static java.util.Arrays.asList;
008import static java.util.UUID.randomUUID;
009import static javax.ws.rs.core.Response.Status.OK;
010import static org.custommonkey.xmlunit.XMLAssert.assertXMLEqual;
011import static org.junit.Assert.assertEquals;
012
013import java.io.File;
014import java.io.IOException;
015import java.nio.file.Path;
016import java.util.List;
017
018import javax.inject.Inject;
019import javax.ws.rs.core.Response;
020
021import org.apache.cxf.jaxrs.client.WebClient;
022import org.junit.Test;
023import org.junit.runner.RunWith;
024import org.springframework.test.context.ContextConfiguration;
025import org.springframework.test.context.junit4.SpringJUnit4ClassRunner;
026import org.xml.sax.SAXException;
027
028import com.google.common.io.Resources;
029
030@RunWith(SpringJUnit4ClassRunner.class)
031@ContextConfiguration("/spring-xml/default-operation.xml")
032public class OperationIT {
033
034    private static final String properResponse;
035    static {
036        try {
037            properResponse =
038                    Resources.toString(new File("src/test/resources/OperationITData")
039                            .toURI().toURL(), UTF_8);
040        } catch (final IOException e) {
041            throw new AssertionError(e);
042        }
043    }
044
045    @Inject
046    private WebClient client;
047
048    @Test
049    public void testUrlOperation() throws IOException, SAXException {
050        final Path dataFileLocation = createTempFile("testUrlOperation", "-" + randomUUID());
051        final List<String> lines = asList("NAME,RANK,SERIAL NUMBER", "Kirk,Admiral,002", "McCoy,Doctor,567");
052        write(dataFileLocation, lines, UTF_8);
053        final Response r = client.query("url", "file://" + dataFileLocation.toAbsolutePath()).get();
054        assertEquals(OK.getStatusCode(), r.getStatus());
055        final String text = r.readEntity(String.class);
056        assertXMLEqual(properResponse, text);
057    }
058}
