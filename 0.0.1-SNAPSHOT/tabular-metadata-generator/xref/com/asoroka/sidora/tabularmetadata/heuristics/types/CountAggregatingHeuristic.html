24  /**
25   * A {@link Heuristic} that aggregates candidate type appearance information for its field.
26   * 
27   * @author ajs6f
28   * @param <SelfType>
29   */
30  public abstract class CountAggregatingHeuristic<SelfType extends CountAggregatingHeuristic<SelfType>> extends
31          ValueCountingHeuristic<SelfType> {
32  
33      private static final ImmutableMap<DataType, Integer> zeroes = toMap(DataType.valuesSet(), constant(0));
34  
35      private static final Logger log = getLogger(CountAggregatingHeuristic.class);
36  
37      /**
38       * In this {@link Map}, we aggregate counts of parseable values for each datatype.
39       */
40      protected EnumMap<DataType, Integer> typeCounts;
41  
42      /**
43       * Initialize counts for each datatype.
44       */
45      @Override
46      public void reset() {
47          super.reset();
48          this.typeCounts = new EnumMap<>(DataType.class);
49          this.typeCounts.putAll(zeroes);
50      }
51  
52      @Override
53      public boolean addValue(final String lex) {
54          log.trace("Accepting lex: {}", lex);
55          if (super.addValue(lex)) {
56              incrementCounts(parseableAs(lex));
57              return true;
58          }
59          return false;
60      }
61  
62      private void incrementCounts(final Collection<DataType> types) {
63          for (final DataType type : types) {
64              typeCounts.put(type, typeCounts.get(type) + 1);
65          }
66      }
67  }
