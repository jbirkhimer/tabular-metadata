26  /**
27   * Calculates the ranges of values supplied for each possible parseable type, without caching the values supplied.
28   * 
29   * @author ajs6f
30   */
31  public class RunningMinMaxHeuristic extends AbstractHeuristic<RunningMinMaxHeuristic> implements
32          RangeDeterminingHeuristic<RunningMinMaxHeuristic> {
33  
34      /**
35       * A {@link Map} from data types to the minimum value from all presented values that were parseable in that type.
36       */
37      protected Map<DataType, Comparable<?>> minimums;
38  
39      /**
40       * A {@link Map} from data types to the maximum value from all presented values that were parseable in that type.
41       */
42      protected Map<DataType, Comparable<?>> maximums;
43  
44      private static final Logger log = getLogger(RunningMinMaxHeuristic.class);
45  
46      /**
47       * Initialize minimums and maximums.
48       */
49      @Override
50      public void reset() {
51          this.minimums = new EnumMap<>(DataType.class);
52          this.maximums = new EnumMap<>(DataType.class);
53      }
54  
55      @Override
56      public boolean addValue(final String value) {
57          for (final DataType type : parseableAs(value)) {
58              final Comparable<?> currentMin = minimums.get(type);
59              final Comparable<?> currentMax = maximums.get(type);
60              try {
61                  final Comparable<?> v = type.parse(value);
62                  log.trace("Trying new value {} against current min {} and current max {} for type {}", v, currentMin,
63                          currentMax, type);
64                  // TODO avoid this repeated conditional
65                  minimums.put(type, (currentMin == null) ? v : natural().min(currentMin, v));
66                  maximums.put(type, (currentMax == null) ? v : natural().max(currentMax, v));
67                  log.trace("Tried new value {} and got new min {} and new max {} for type {}", v, minimums.get(type),
68                          maximums.get(type), type);
69  
70              } catch (final ParsingException e) {
71                  // we are only parsing for types that have already been checked
72                  throw new AssertionError("Could not parse to a type that was passed as parsing!", e);
73              }
74          }
75          return true;
76      }
77  
78      @Override
79      public Map<DataType, Range<?>> getRanges() {
80          return asMap(DataType.valuesSet(), getRangeForType());
81      }
82  
83      private Function<DataType, Range<?>> getRangeForType() {
84          return new Function<DataType, Range<?>>() {
85  
86              @Override
87              public Range<?> apply(final DataType type) {
88                  final boolean hasMin = minimums.containsKey(type);
89                  final boolean hasMax = maximums.containsKey(type);
90                  final Comparable<?> min = minimums.get(type);
91                  final Comparable<?> max = maximums.get(type);
92                  if (hasMin) {
93                      if (hasMax) {
94                          return Range.closed(min, max);
95                      }
96                      return atLeast(min);
97                  }
98                  if (hasMax) {
99                      return atMost(max);
100                 }
101                 return Range.all();
102             }
103         };
104     }
105 
106     @Override
107     public RunningMinMaxHeuristic newInstance() {
108         return new RunningMinMaxHeuristic();
109     }
110 
111     @Override
112     public RunningMinMaxHeuristic get() {
113         return this;
114     }
