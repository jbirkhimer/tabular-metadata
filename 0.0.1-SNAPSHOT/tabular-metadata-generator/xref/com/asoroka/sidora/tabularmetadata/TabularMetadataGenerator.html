41  /**
42   * Master entry point for this API. This is the class from which all parsing should initiate, via
43   * {@link #getMetadata(URL)}.
44   * 
45   * @author ajs6f
46   */
47  public class TabularMetadataGenerator {
48  
49      private static final Charset CHARACTER_ENCODING = UTF_8;
50  
51      private CSVFormat format = DEFAULT;
52  
53      /**
54       * Default value of {@code 0} indicates no limit. See {@link TabularScanner.scan(final int limit)}.
55       */
56      private Integer scanLimit = 0;
57  
58      private RangeDeterminingHeuristic<?> rangeStrategy = new RunningMinMaxHeuristic();
59  
60      private TypeDeterminingHeuristic<?> typeStrategy = new StrictHeuristic();
61  
62      private EnumeratedValuesHeuristic<?> enumStrategy = new InMemoryEnumeratedValuesHeuristic();
63  
64      private HeaderHeuristic<?> headerStrategy = new DefaultHeaderHeuristic();
65  
66      /**
67       * The main entry point to application workflow.
68       * 
69       * @param dataUrl Where to find some tabular data.
70       * @return The results of metadata extraction.
71       * @throws IOException
72       */
73      public TabularMetadata getMetadata(final URL dataUrl) throws IOException {
74  
75          try (final CSVParser csvParser = parse(dataUrl, CHARACTER_ENCODING, format)) {
76              final PeekingIterator<CSVRecord> parser = peekingIterator(csvParser.iterator());
77              // TODO allow a HeaderHeuristic to use more information than the first line of data
78              final CSVRecord firstLine = parser.peek();
79              for (final String field : firstLine) {
80                  headerStrategy.addValue(field);
81              }
82              final boolean hasHeaders = headerStrategy.results();
83              headerStrategy.reset();
84              final List<String> headerNames;
85              if (hasHeaders) {
86                  headerNames = newArrayList(firstLine);
87                  parser.next();
88              }
89              else {
90                  headerNames = emptyHeaders(firstLine.size());
91              }
92              // scan values up to the limit
93              final TabularScanner scanner = new TabularScanner(parser, typeStrategy, rangeStrategy, enumStrategy);
94              scanner.scan(scanLimit);
95  
96              final List<TypeDeterminingHeuristic<?>> typeStrategies = scanner.getTypeStrategies();
97              final List<RangeDeterminingHeuristic<?>> rangeStrategies = scanner.getRangeStrategies();
98              final List<EnumeratedValuesHeuristic<?>> enumStrategies = scanner.getEnumStrategies();
99  
100             // extract the results for each field
101             final List<SortedSet<DataType>> columnTypes =
102                     sequence(typeStrategies).map(new Extract<SortedSet<DataType>>()).toList();
103             final List<Map<DataType, Range<?>>> minMaxes =
104                     sequence(rangeStrategies).map(new Extract<Map<DataType, Range<?>>>()).toList();
105             final List<Map<DataType, Set<String>>> enumValues =
106                     sequence(enumStrategies).map(new Extract<Map<DataType, Set<String>>>()).toList();
107 
108             return new TabularMetadata(headerNames, columnTypes, minMaxes, enumValues);
109         } catch (final NoSuchElementException e) {
110             throw new EmptyDataFileException(dataUrl + " has no data in it!");
111         }
112     }
113 
114     private static final List<String> emptyHeaders(final int numFields) {
115         final List<String> headers = new ArrayList<>(numFields);
116         for (int i = 1; i <= numFields; i++) {
117             headers.add("Variable " + i);
118         }
119         return headers;
120     }

122     /**
123      * @param scanLimit A limit to the number of rows to scan, including any header row that may be present. {@code 0}

126     public void setScanLimit(final Integer scanLimit) {
