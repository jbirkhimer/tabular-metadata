6   import java.util.Comparator;
7   import java.util.List;
8   import java.util.Map;
9   import java.util.NavigableMap;
10  import java.util.Set;
11  import java.util.SortedSet;
12  import java.util.TreeMap;
13  
14  import com.asoroka.sidora.tabularmetadata.datatype.DataType;
15  import com.google.common.base.MoreObjects;
16  import com.google.common.base.MoreObjects.ToStringHelper;
17  import com.google.common.collect.Range;
18  import com.googlecode.totallylazy.Function1;
19  import com.googlecode.totallylazy.Pair;
20  
21  /**
22   * A container for the results of metadata extraction on a single data file, with machinery to present those results
23   * in appropriate orders.
24   * 
25   * @author ajs6f
26   */
27  public class TabularMetadata {
28  
29      private final List<String> headerNames;
30  
31      private final List<SortedSet<DataType>> fieldTypes;
32  
33      private final List<Map<DataType, Range<?>>> minMaxes;
34  
35      private final List<Map<DataType, Set<String>>> enumeratedValues;
36  
37      /**
38       * @param headerNames
39       * @param fieldTypes
40       * @param minMaxes
41       * @param enumeratedValues
42       */
43      public TabularMetadata(final List<String> headerNames, final List<SortedSet<DataType>> fieldTypes,
44              final List<Map<DataType, Range<?>>> minMaxes, final List<Map<DataType, Set<String>>> enumeratedValues) {
45          this.headerNames = headerNames;
46          this.fieldTypes = fieldTypes;
47          this.minMaxes = minMaxes;
48          this.enumeratedValues = enumeratedValues;
49      }
50  
51      /**
52       * Sorts a {@link Map<DataType,T>} in the same way that the corresponding field in {@link #fieldTypes()} is
53       * sorted. WARNING: Must not be used before setting {@link #fieldTypes}!
54       */
55      private <T> Function1<Pair<Number, Map<DataType, T>>, NavigableMap<DataType, T>> sortByLikelihood() {
56          return new Function1<Pair<Number, Map<DataType, T>>, NavigableMap<DataType, T>>() {
57  
58              @Override
59              public NavigableMap<DataType, T> call(final Pair<Number, Map<DataType, T>> minMaxMapWithIndex) {
60                  final Comparator<? super DataType> sortByLikelihood =
61                          fieldTypes().get(minMaxMapWithIndex.getKey().intValue()).comparator();
62                  final TreeMap<DataType, T> sortedByLikelihood = new TreeMap<>(sortByLikelihood);
63                  sortedByLikelihood.putAll(minMaxMapWithIndex.getValue());
64                  return sortedByLikelihood;
65              }
66          };
67      }
68  
69      /**
70       * @return A list of prospective header names.
71       */
72      public List<String> headerNames() {
73          return headerNames;
74      }
75  
76      /**
77       * @return A list of candidates for field datatypes, one candidate set per field. Each candidate set is sorted
78       *         according to a notion of decreasing plausibility that is specific to the type-determination strategy
79       *         used.
80       */
81      public List<SortedSet<DataType>> fieldTypes() {
82          return fieldTypes;
83      }
84  
85      /**
86       * @return A list of maps from {@link DataType}s to {@link Range}s, with each data type mapped to the minimum and
87       *         maximum for each field, <i>if</i> that field is treated as of that type. The maps are sorted in the
88       *         same manner as is {@link #fieldTypes()}. The values of the endpoints of the ranges, if they exist, are
89       *         in the Java type value-space associated to the datatype considered most likely for that field. The idea
90       *         here is that when a given field is finally determined to be of a given type (by user action after
91       *         automatic action), the appropriate range can be looked up at that time. This is to ensure that in a
92       *         situation where the type determination strategy employed gave a wrong answer, the correct answer for
93       *         range can still be found after the type determination has been corrected.
94       * @see com.google.common.collect.Range<?>
95       */
96      public List<NavigableMap<DataType, Range<?>>> minMaxes() {
97          return sequence(minMaxes).zipWithIndex().map(this.<Range<?>> sortByLikelihood()).toList();
98      }
99  
100     /**
101      * @return A list (one element for each field in the input file) of maps from each possible datatype to an set of
102      *         the lexes found in that field that were parseable into that data type. The maps are sorted in the same
103      *         manner as is {@link #fieldTypes()}.
104      */
105     public List<NavigableMap<DataType, Set<String>>> enumeratedValues() {
106         return sequence(enumeratedValues).zipWithIndex().map(this.<Set<String>> sortByLikelihood()).toList();
107     }
108 
109     private static final ToStringHelper toStringHelper() {
110         return MoreObjects.toStringHelper(TabularMetadata.class);
111     }
112 
113     @Override
114     public String toString() {
115         return toStringHelper().add("headerNames", headerNames()).add("fieldTypes", fieldTypes())
116                 .add("enumeratedValues", enumeratedValues()).add("minMaxes", minMaxes()).toString();
117     }
118 }
