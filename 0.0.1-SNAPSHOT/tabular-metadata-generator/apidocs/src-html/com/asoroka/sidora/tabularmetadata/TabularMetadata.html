002package com.asoroka.sidora.tabularmetadata;
003
004import java.util.List;
005import java.util.Map;
006import java.util.Set;
007import java.util.SortedSet;
008
009import com.asoroka.sidora.tabularmetadata.datatype.DataType;
010import com.google.common.collect.Range;
011
012/**
013 * A container for the results of metadata extraction on a single data file.
014 * 
015 * @author ajs6f
016 */
017public class TabularMetadata {
018
019    private final List<String> headerNames;
020
021    private final List<SortedSet<DataType>> fieldTypes;
022
023    private final List<Map<DataType, Range<?>>> minMaxes;
024
025    private final List<Map<DataType, Set<String>>> enumeratedValues;
026
027    /**
028     * @param headerNames
029     * @param fieldTypes
030     * @param minMaxes
031     * @param enumeratedValues
032     */
033    public TabularMetadata(final List<String> headerNames, final List<SortedSet<DataType>> fieldTypes,
034            final List<Map<DataType, Range<?>>> minMaxes, final List<Map<DataType, Set<String>>> enumeratedValues) {
035        this.headerNames = headerNames;
036        this.fieldTypes = fieldTypes;
037        this.minMaxes = minMaxes;
038        this.enumeratedValues = enumeratedValues;
039    }
040
041    /**
042     * @return A list of header names, empty if there was no header row found.
043     */
044    public List<String> headerNames() {
045        return headerNames;
046    }
047
048    /**
049     * @return A list of candidates for field datatypes, one candidate set per field. Each candidate set is sorted
050     *         according to a notion of decreasing plausibility that is specific to the type-determination strategy
051     *         used.
052     */
053    public List<SortedSet<DataType>> fieldTypes() {
054        return fieldTypes;
055    }
056
057    /**
058     * @return A list of maps from {@link DataType}s to {@link Range}s, with each data type mapped to the minimum and
059     *         maximum for each field, <i>if</i> that field is treated as correctly of that type. The values of the
060     *         endpoints of the ranges, if they exist, are in the Java type value-space associated to the datatype
061     *         considered most likely for that field. The idea here is that when a given field is finally determined
062     *         to be of a given type (by user action after automatic action), the appropriate range can be looked up
063     *         at that time. This is to ensure that in a situation where the type determination strategy employed gave
064     *         a wrong answer, the correct answer for range can still be found after the type determination has been
065     *         corrected.
066     * @see com.google.common.collect.Range<?>
067     */
068    public List<Map<DataType, Range<?>>> minMaxes() {
069        return minMaxes;
070    }
071
072    /**
073     * @return a list (one element for each field in the input file) of maps from each possible datatype to an set of
074     *         the lexes found in that field that were parseable into that data type.
075     */
076    public List<Map<DataType, Set<String>>> enumeratedValues() {
077        return enumeratedValues;
078    }
079}
