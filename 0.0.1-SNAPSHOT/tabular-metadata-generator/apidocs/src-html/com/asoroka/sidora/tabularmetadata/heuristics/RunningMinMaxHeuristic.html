006package com.asoroka.sidora.tabularmetadata.heuristics;
007
008import static com.asoroka.sidora.tabularmetadata.datatype.DataType.parseableAs;
009import static com.google.common.collect.Ordering.natural;
010
011import java.util.EnumMap;
012import java.util.Map;
013
014import com.asoroka.sidora.tabularmetadata.datatype.DataType;
015import com.asoroka.sidora.tabularmetadata.datatype.ParsingException;
016import com.google.common.collect.Range;
017
018/**
019 * Calculates the ranges of values supplied for each possible parseable type, without caching the values supplied.
020 * 
021 * @author ajs6f
022 * @param <T>
023 */
024public abstract class RunningMinMaxHeuristic<T extends RunningMinMaxHeuristic<T>> implements
025        DataTypeHeuristic<T> {
026
027    /**
028     * A {@link Map} from data types to the minimum value from all presented values that were parseable in that type.
029     */
030    private Map<DataType, Comparable<?>> minimums = new EnumMap<>(DataType.class);
031
032    /**
033     * A {@link Map} from data types to the maximum value from all presented values that were parseable in that type.
034     */
035    protected Map<DataType, Comparable<?>> maximums = new EnumMap<>(DataType.class);
036
037    @Override
038    public void addValue(final String input) {
039        for (final DataType type : parseableAs(input)) {
040            final Comparable<?> currentMin = minimums.get(type);
041            final Comparable<?> currentMax = maximums.get(type);
042            try {
043                final Comparable<?> value = type.parse(input);
044                // TODO avoid this repeated conditional
045                minimums.put(type, (currentMin == null) ? value : natural().min(currentMin, value));
046                maximums.put(type, (currentMax == null) ? value : natural().max(currentMax, value));
047            } catch (final ParsingException e) {
048                // NO OP
049                // we don't care if a parse attempt fails because we're garnering measures for those that succeed
050            }
051        }
052    }
053
054    @Override
055    public <MinMax extends Comparable<MinMax>> Range<MinMax> getRange() {
056        final MinMax min = (MinMax) minimums.get(mostLikelyType());
057        final MinMax max = (MinMax) maximums.get(mostLikelyType());
058        return Range.closed(min, max);
059    }
060
061    @Override
062    abstract public T clone();
063
064}
