002package com.asoroka.sidora.tabularmetadata.heuristics.types;
003
004import static com.asoroka.sidora.tabularmetadata.datatype.DataType.sortByHierarchy;
005import static com.google.common.collect.Sets.filter;
006import static org.slf4j.LoggerFactory.getLogger;
007
008import java.util.Set;
009import java.util.SortedSet;
010
011import org.slf4j.Logger;
012
013import com.asoroka.sidora.tabularmetadata.datatype.DataType;
014import com.google.common.base.Predicate;
015
016/**
017 * A {@link TypeDeterminingHeuristic} that uses some test that maps types to boolean acceptance values. The test is
018 * created by overriding {@link #candidacy(DataType)}. Types are passed or rejected without any particular order other
019 * than specificity of type within the hierarchy.
020 * 
021 * @author ajs6f
022 * @param <SelfType>
023 */
024public abstract class PerTypeHeuristic<SelfType extends PerTypeHeuristic<SelfType> & TypeDeterminingHeuristic<SelfType>>
025        extends CountAggregatingHeuristic<SelfType, SortedSet<DataType>> implements
026        TypeDeterminingHeuristic<SelfType> {
027
028    private static final Logger log = getLogger(PerTypeHeuristic.class);
029
030    @Override
031    public SortedSet<DataType> results() {
032        // develop a set of candidate types in a manner specific to the subclass
033        final Set<DataType> possibleTypes = filter(DataType.valuesSet(), candidacyPredicate);
034        // order by hierarchy
035        final SortedSet<DataType> sortedCandidates = sortByHierarchy(possibleTypes);
036        log.trace("Found candidate types: {}", sortedCandidates);
037
038        return sortedCandidates;
039    }
040
041    private Predicate<DataType> candidacyPredicate = new Predicate<DataType>() {
042
043        @Override
044        public boolean apply(final DataType type) {
045            return candidacy(type);
046        }
047    };
048
049    /**
050     * Subclasses must override this method with an algorithm that uses the gathered statistics (and possibly other
051     * information) to make a determination about the most likely type of the proffered values.
052     * 
053     * @return Whether this type should be considered as a candidate for selection.
054     */
055    protected abstract boolean candidacy(final DataType type);
056
057    @Override
058    public abstract SelfType newInstance();
059}
