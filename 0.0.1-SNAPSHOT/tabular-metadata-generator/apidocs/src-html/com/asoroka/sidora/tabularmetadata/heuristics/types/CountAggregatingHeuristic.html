006package com.asoroka.sidora.tabularmetadata.heuristics.types;
007
008import static com.asoroka.sidora.tabularmetadata.datatype.DataType.parseableAs;
009import static com.google.common.base.Functions.constant;
010import static com.google.common.collect.Maps.toMap;
011import static org.slf4j.LoggerFactory.getLogger;
012
013import java.util.Collection;
014import java.util.EnumMap;
015import java.util.Map;
016
017import org.slf4j.Logger;
018
019import com.asoroka.sidora.tabularmetadata.datatype.DataType;
020import com.asoroka.sidora.tabularmetadata.heuristics.Heuristic;
021import com.asoroka.sidora.tabularmetadata.heuristics.ValueCountingHeuristic;
022import com.google.common.collect.ImmutableMap;
023
024/**
025 * A {@link Heuristic} that aggregates candidate type appearance information for its field.
026 * 
027 * @author ajs6f
028 * @param <SelfType>
029 */
030public abstract class CountAggregatingHeuristic<SelfType extends CountAggregatingHeuristic<SelfType, ResultType>, ResultType>
031        extends ValueCountingHeuristic<SelfType, ResultType> {
032
033    private static final ImmutableMap<DataType, Integer> zeroes = toMap(DataType.valuesSet(), constant(0));
034
035    private static final Logger log = getLogger(CountAggregatingHeuristic.class);
036
037    /**
038     * In this {@link Map}, we aggregate counts of parseable values for each datatype.
039     */
040    protected EnumMap<DataType, Integer> typeCounts;
041
042    /**
043     * Initialize counts for each datatype.
044     */
045    @Override
046    public void reset() {
047        super.reset();
048        this.typeCounts = new EnumMap<>(DataType.class);
049        this.typeCounts.putAll(zeroes);
050    }
051
052    @Override
053    public boolean addValue(final String lex) {
054        log.trace("Accepting lex: {}", lex);
055        if (super.addValue(lex)) {
056            incrementCounts(parseableAs(lex));
057            return true;
058        }
059        return false;
060    }
061
062    private void incrementCounts(final Collection<DataType> types) {
063        for (final DataType type : types) {
064            typeCounts.put(type, typeCounts.get(type) + 1);
065        }
066    }
067}
