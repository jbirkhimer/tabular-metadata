002package com.asoroka.sidora.tabularmetadata;
003
004import javax.inject.Provider;
005
006/**
007 * @author ajs6f
008 */
009public interface CloneableProvider<SelfType extends CloneableProvider<SelfType>> extends java.lang.Cloneable,
010        Provider<SelfType> {
011
012    /**
013     * We override {@link Object#clone()} in order to narrow its return type for type-safety in the use of this
014     * method.
015     * 
016     * @see java.lang.Object#clone()
017     * @return A clone of this object.
018     */
019    public SelfType clone();
020
021}
