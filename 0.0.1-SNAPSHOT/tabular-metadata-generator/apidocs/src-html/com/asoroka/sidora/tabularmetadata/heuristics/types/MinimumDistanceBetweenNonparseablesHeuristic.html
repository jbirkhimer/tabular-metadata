002package com.asoroka.sidora.tabularmetadata.heuristics.types;
003
004import static com.asoroka.sidora.tabularmetadata.datatype.DataType.parseableAs;
005import static com.google.common.base.Functions.constant;
006import static com.google.common.collect.Maps.toMap;
007import static java.lang.Float.NEGATIVE_INFINITY;
008import static java.util.EnumSet.complementOf;
009
010import java.util.EnumMap;
011import java.util.EnumSet;
012import java.util.Map;
013
014import com.asoroka.sidora.tabularmetadata.datatype.DataType;
015
016public class MinimumDistanceBetweenNonparseablesHeuristic extends
017        PerTypeHeuristic<MinimumDistanceBetweenNonparseablesHeuristic> {
018
019    private final int minimumDistance;
020
021    private Map<DataType, Boolean> candidateTypes;
022
023    /**
024     * We define locations as {@link Float}s in order to use the special value {@link Float#NEGATIVE_INFINITY}, which
025     * does not exist for integer types in Java.
026     */
027    private Map<DataType, Float> locationsOfLastNonparseables;
028
029    public MinimumDistanceBetweenNonparseablesHeuristic(final int minimumDistance) {
030        super();
031        this.minimumDistance = minimumDistance;
032    }
033
034    @Override
035    public void reset() {
036        super.reset();
037        candidateTypes = new EnumMap<>(DataType.class);
038        locationsOfLastNonparseables = new EnumMap<>(DataType.class);
039        // assume that every type is a candidate
040        final Map<DataType, Boolean> allTrue = toMap(DataType.valuesSet(), constant(true));
041        candidateTypes.putAll(allTrue);
042
043        // record that we haven't yet seen any nonparseables
044        final Map<DataType, Float> originalLocations = toMap(DataType.valuesSet(), constant(NEGATIVE_INFINITY));
045        locationsOfLastNonparseables.putAll(originalLocations);
046    }
047
048    @Override
049    protected boolean candidacy(final DataType type) {
050        return candidateTypes.get(type);
051    }
052
053    @Override
054    public boolean addValue(final String value) {
055        if (super.addValue(value)) {
056            for (final DataType type : notParseableAs(value)) {
057                final float distanceToLastNonParseableOfThisType =
058                        totalNumValues() - locationsOfLastNonparseables.get(type);
059                if (distanceToLastNonParseableOfThisType < minimumDistance) {
060                    // it's been too soon since the last nonparseable value of this type, knock it out of the running
061                    candidateTypes.put(type, false);
062                } else {
063                    // mark that we saw a nonparseable value for this type
064                    locationsOfLastNonparseables.put(type, (float) totalNumValues());
065                }
066            }
067            return true;
068        }
069        return false;
070    }
071
072    /**
073     * @param value
074     * @return an {@link EnumSet} of those DataTypes into which s cannot be parsed
075     */
076    private static EnumSet<DataType> notParseableAs(final String value) {
077        return complementOf(parseableAs(value));
078    }
079
080    @Override
081    public MinimumDistanceBetweenNonparseablesHeuristic newInstance() {
082        return new MinimumDistanceBetweenNonparseablesHeuristic(minimumDistance);
083    }
084
085    @Override
086    public MinimumDistanceBetweenNonparseablesHeuristic get() {
087        return this;
088    }
089
090}
