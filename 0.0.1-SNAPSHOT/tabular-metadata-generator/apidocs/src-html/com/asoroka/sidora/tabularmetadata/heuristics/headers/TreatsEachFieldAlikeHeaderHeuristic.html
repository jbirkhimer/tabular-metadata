004import static com.google.common.collect.Iterables.all;
005import static org.slf4j.LoggerFactory.getLogger;
006
007import java.util.ArrayList;
008import java.util.List;
009
010import org.slf4j.Logger;
011
012import com.asoroka.sidora.tabularmetadata.heuristics.AbstractHeuristic;
013import com.google.common.base.Predicate;
014
015/**
016 * As the name implies, a {@link HeaderHeuristic} that treats each field in the row the same and applies a single test
017 * to each. Only if every field value passes the field-test does the row pass this test. Subclasses define the test by
018 * implementing {@link #fieldTest()}.
019 * 
020 * @author ajs6f
021 * @param <SelfType>
022 */
023public abstract class TreatsEachFieldAlikeHeaderHeuristic<SelfType extends TreatsEachFieldAlikeHeaderHeuristic<SelfType>>
024        extends AbstractHeuristic<SelfType> implements HeaderHeuristic<SelfType> {
025
026    private static final Logger log = getLogger(TreatsEachFieldAlikeHeaderHeuristic.class);
027
028    private List<String> inputRow;
029
030    @Override
031    public boolean isHeader() {
032        log.trace("Checking input row: {}", inputRow);
033        return all(inputRow, fieldTest());
034    }
035
036    protected abstract Predicate<? super String> fieldTest();
037
038    @Override
039    public boolean addValue(final String value) {
040        return inputRow.add(value);
041    }
042
043    @Override
044    public void reset() {
045        inputRow = new ArrayList<>();
046    }
047
048    @Override
049    public abstract SelfType clone();
050}
