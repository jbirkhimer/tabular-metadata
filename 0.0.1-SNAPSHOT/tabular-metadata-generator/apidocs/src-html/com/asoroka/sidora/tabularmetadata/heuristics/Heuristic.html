002package com.asoroka.sidora.tabularmetadata.heuristics;
003
004import com.asoroka.sidora.tabularmetadata.SelfTypeInstanceGenerator;
005import com.googlecode.totallylazy.Function1;
006
007/**
008 * The simple idea of a heuristic that accepts lexes and does some kind of work with them. See its subtypes for
009 * examples of use.
010 * 
011 * @author ajs6f
012 */
013public interface Heuristic<SelfType extends Heuristic<SelfType, ResultType>, ResultType> extends
014        SelfTypeInstanceGenerator<SelfType> {
015
016    /**
017     * Provide a value to this heuristic for consideration.
018     * 
019     * @param lex the value to consider
020     * @return whether or not to continue evaluating the lex (for use in polymorphic chains of evaluation)
021     */
022    boolean addValue(final String lex);
023
024    /**
025     * Resets this heuristic to discard all gathered information.
026     */
027    void reset();
028
029    ResultType results();
030
031    public static class Extract<ResultType> extends Function1<Heuristic<?, ResultType>, ResultType> {
032
033        @Override
034        public ResultType call(final Heuristic<?, ResultType> strategy) {
035            return strategy.results();
036        }
037    }
038}
