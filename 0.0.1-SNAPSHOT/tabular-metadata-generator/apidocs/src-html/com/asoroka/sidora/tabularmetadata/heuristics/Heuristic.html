002package com.asoroka.sidora.tabularmetadata.heuristics;
003
004import com.asoroka.sidora.tabularmetadata.SelfTypeInstanceGenerator;
005
006/**
007 * The simple idea of a heuristic that accepts lexes and does some kind of work with them. See its subtypes for
008 * examples of use.
009 * 
010 * @author ajs6f
011 */
012public interface Heuristic<SelfType extends Heuristic<SelfType>> extends SelfTypeInstanceGenerator<SelfType> {
013
014    /**
015     * Provide a value to this heuristic for consideration.
016     * 
017     * @param lex the value to consider
018     * @return whether or not to continue evaluating the lex (for use in polymorphic chains of evaluation)
019     */
020    boolean addValue(final String lex);
021
022    /**
023     * Resets this heuristic to discard all gathered information.
024     */
025    void reset();
026
027}
