006package com.asoroka.sidora.tabularmetadata.heuristics;
007
008import static com.asoroka.sidora.tabularmetadata.datatype.DataType.String;
009import static com.asoroka.sidora.tabularmetadata.datatype.DataType.firstMostRestrictiveType;
010import static com.google.common.collect.Sets.filter;
011import static java.util.Objects.hash;
012import static org.slf4j.LoggerFactory.getLogger;
013
014import java.util.EnumMap;
015import java.util.Map;
016import java.util.Set;
017
018import org.slf4j.Logger;
019
020import com.asoroka.sidora.tabularmetadata.datatype.DataType;
021import com.google.common.base.Predicate;
022
023/**
024 * A {@link DataTypeHeuristic} that aggregates candidate type appearance information for its field.
025 * 
026 * @author ajs6f
027 * @param <T>
028 */
029public abstract class CountAggregatingHeuristic<T extends CountAggregatingHeuristic<T>> extends
030        RunningMinMaxHeuristic<CountAggregatingHeuristic<T>> {
031
032    /**
033     * In this {@link Map}, we aggregate counts of parseable values for each datatype.
034     */
035    protected final EnumMap<DataType, Integer> typeCounts = new EnumMap<>(DataType.class);
036
037    private static final Logger log = getLogger(CountAggregatingHeuristic.class);
038
039    /**
040     * Initialize counts for each datatype.
041     */
042    public CountAggregatingHeuristic() {
043        for (final DataType type : DataType.values()) {
044            typeCounts.put(type, 0);
045        }
046    }
047
048    @Override
049    public DataType mostLikelyType() {
050        log.trace("Working with {} total values.", totalNumValues());
051        // we filter to only those types which were selected as parseable for all values
052        final Set<DataType> possibleTypes = filter(typeCounts.keySet(), new Predicate<DataType>() {
053
054            @Override
055            public boolean apply(final DataType type) {
056                return candidacy(type);
057            }
058        });
059        log.trace("Found candidate types: {}", possibleTypes);
060        return firstMostRestrictiveType(possibleTypes);
061
062    }
063
064    /**
065     * We make use here of the fact that {@link DataType#String} is the top type, so anything that can be parsed can
066     * be parsed as a String.
067     * 
068     * @return The total number of values seen so far
069     */
070    protected int totalNumValues() {
071        return typeCounts.get(String);
072    }
073
074    /**
075     * @return Whether this type should be considered as a candidate for selection.
076     */
077    abstract protected boolean candidacy(DataType d);
078
079    @Override
080    public void addValue(final String value) {
081        super.addValue(value);
082        for (final DataType type : DataType.parseableAs(value)) {
083            typeCounts.put(type, typeCounts.get(type) + 1);
084        }
085    }
086
087    @Override
088    abstract public T clone();
089
090    @Override
091    public int hashCode() {
092        return hash(typeCounts);
093    }
094
095    @Override
096    public boolean equals(final Object o) {
097
098        if (!(CountAggregatingHeuristic.class.isInstance(o))) {
099            return false;
100        }
101        final CountAggregatingHeuristic<?> oo = (CountAggregatingHeuristic<?>) o;
102        return this.hashCode() == oo.hashCode();
103    }
104}
