002package com.asoroka.sidora.tabularmetadata.heuristics.types;
003
004import java.util.SortedSet;
005
006import com.asoroka.sidora.tabularmetadata.datatype.DataType;
007import com.asoroka.sidora.tabularmetadata.heuristics.Heuristic;
008
009/**
010 * Determines into which {@link DataType}s a series of values most likely falls.
011 * 
012 * @author ajs6f
013 */
014public interface TypeDeterminingHeuristic<SelfType extends TypeDeterminingHeuristic<SelfType>> extends
015        Heuristic<SelfType> {
016
017    /**
018     * @return Types for the proffered values in order of their likelihood according to this heuristic.
019     */
020    public SortedSet<DataType> typesAsLikely();
021
022    /**
023     * @return The single most likely type for the proffered values according to this heuristic. Under any normal
024     *         regime, this should be equal to {@code typesAsLikely().first()}.
025     */
026    public DataType mostLikelyType();
027
028}
