035/**
036 * This enumeration constructs
037 * <ul>
038 * <li>our "native" type system, as well as</li>
039 * <li>maps from lexical spaces in our system to selected portions of selected value spaces in the Java type system,
040 * and</li>
041 * <li>maps from our types to types in the XSD type system.</li>
042 * </ul>
043 * It also offers convenience methods for working with our type system.
044 * 
045 * @author ajs6f
046 */
047public enum DataType {
048
049    String(W3C_XML_SCHEMA_NS_URI + "string", null) {
050
051        /*
052         * This override indicates that String is the top type.
053         */
054        @Override
055        public Set<DataType> supertypes() {
056            return EnumSet.of((DataType) this);
057        }
058
059        @SuppressWarnings("unchecked")
060        @Override
061        public String parse(final java.lang.String s) {
062            return s;
063        }
064
065    },
066    Decimal(W3C_XML_SCHEMA_NS_URI + "decimal", String) {
067
068        @SuppressWarnings("unchecked")
069        @Override
070        public Float parse(final java.lang.String s) throws ParsingException {
071            try {
072                return parseFloat(s);
073            } catch (final NumberFormatException e) {
074                throw new ParsingException("Could not parse as Decimal!", e);
075            }
076        }
077    },
078    Integer(W3C_XML_SCHEMA_NS_URI + "integer", Decimal) {
079
080        @SuppressWarnings("unchecked")
081        @Override
082        public Integer parse(final java.lang.String s) throws ParsingException {
083            try {
084                return parseInt(s);
085            } catch (final NumberFormatException e) {
086                throw new ParsingException("Could not parse as Integer!", e);
087            }
088        }
089    },
090    NonNegativeInteger(W3C_XML_SCHEMA_NS_URI + "nonNegativeInteger", Integer) {
091
092        @SuppressWarnings("unchecked")
093        @Override
094        public Integer parse(final java.lang.String s) throws ParsingException {
095
096            final Integer value = Integer.parse(s);
097            if (value > -1) {
098                return value;
099            }
100            throw new ParsingException("Value was negative!");
101        }
102    },
103    PositiveInteger(W3C_XML_SCHEMA_NS_URI + "positiveInteger", NonNegativeInteger) {
104
105        @SuppressWarnings("unchecked")
106        @Override
107        public Integer parse(final java.lang.String s) throws ParsingException {
108            final Integer value = NonNegativeInteger.parse(s);
109            if (value > 0) {
110                return value;
111            }
112            throw new ParsingException("Value was negative!");
113        }
114    },
115    Geographic(null, String) {
116
117        @SuppressWarnings("unchecked")
118        @Override
119        public GeographicValue parse(final String s) throws ParsingException {
120            try {
121                final List<Float> parts =
122                        transform(asList(s.split(",")), string2float);
123                return new GeographicValue(parts);
124            } catch (final IllegalArgumentException e) {
125                throw new ParsingException("Could not parse as Geographic!", e);
126            }
127        }
128    },
129    Boolean(W3C_XML_SCHEMA_NS_URI + "boolean", String) {
130
131        @SuppressWarnings("unchecked")
132        @Override
133        public Boolean parse(final java.lang.String s) throws ParsingException {
134            if (BOOLEAN_TRUE.matcher(s).matches()) {
135                return true;
136            }
137            if (BOOLEAN_FALSE.matcher(s).matches()) {
138                return false;
139            }
140            throw new ParsingException("Could not parse as Boolean!");
141        }
142    },
143    DateTime(W3C_XML_SCHEMA_NS_URI + "dateTime", String) {
144
145        @SuppressWarnings("unchecked")
146        @Override
147        public DateTime parse(final java.lang.String s) throws ParsingException {
148            try {
149                return dateTimeParser().parseDateTime(s);
150            } catch (final IllegalArgumentException e) {
151                throw new ParsingException("Could not parse as DataTime!", e);
152            }
153        }
154    };
155
156    private DataType(final String uri, final DataType supertype) {
157        this.xsdType = uri == null ? null : URI.create(uri);
158        this.supertype = supertype;
159    }
160
161    /**
162     * The XSD type, if any, with which this DataType is associated
163     */
164    public final URI xsdType;
165
166    /**
167     * This DataType, recorded because Java doesn't handle lexical "this" as we might like.
168     */
169    final DataType self = this;
170
171    /**
172     * The immediate supertype of this DataType.
173     */
174    final DataType supertype;
175
176    /**
177     * A memoized form of the supertypes of this DataType, used to avoid redoing this recursive calculation during
178     * operation.
179     */
180    private final Supplier<Set<DataType>> supertypesSupplier = memoize(new Supplier<Set<DataType>>() {
181
182        @Override
183        public Set<DataType> get() {
184            return EnumSet.of(self, EnumSet.of(supertype, supertype.supertypes().toArray(new DataType[0]))
185                    .toArray(
186                            new DataType[0]));
187        }
188    });
189
190    /**
191     * @return The supertypes of this type, including itself.
192     */
193    public Set<DataType> supertypes() {
194        return supertypesSupplier.get();
195    }
196
197    /**
198     * Map from lex to Java value.
199     * 
200     * @param s
201     * @return s parsed into an instance of a Java type representing this {@link DataType}
202     * @throws ParsingException
203     */
204    abstract public <T extends Comparable<T>> T parse(final String s) throws ParsingException;
205
206    /**
207     * @param s A value to parse
208     * @return a {@link Set} of those DataTypes into which s can be parsed
209     */
210    public static Set<DataType> parseableAs(final String s) {
211        return filter(allOf(DataType.class), new Predicate<DataType>() {
212
213            @Override
214            public boolean apply(final DataType t) {
215                try {
216                    t.parse(s.trim());
217                    return true;
218                } catch (final ParsingException e) {
219                    return false;
220                }
221            }
222        });
223    }
224
225    /**
226     * An ordering by type hierarchy. Those types with more supertypes are considered "larger" than those with fewer.
227     */
228    public static final Comparator<DataType> orderingByHierarchy = new Comparator<DataType>() {
229
230        @Override
231        public int compare(final DataType left, final DataType right) {
232            return left.supertypes().size() - right.supertypes().size();
233        }
234    };
235
236    /**
237     * @param types The types to compare.
238     * @return The first type with as "bottom-most" as position in the hierarchy as any other.
239     */
240    public static DataType firstMostRestrictiveType(final Collection<DataType> types) {
241        return max(types, orderingByHierarchy);
242    }
243
244    // private static final Logger log = getLogger(DataType.class);
245
246    /**
247     * How to recognize a Boolean lex for true.
248     */
249    public static final Pattern BOOLEAN_TRUE = compile("true|t", CASE_INSENSITIVE);
250
251    /**
252     * How to recognize a Boolean lex for false.
253     */
254    public static final Pattern BOOLEAN_FALSE = compile("false|f", CASE_INSENSITIVE);
255
256    static final Function<java.lang.String, Float> string2float = new Function<String, Float>() {
257
258        @Override
259        public Float apply(final java.lang.String seg) {
260            return parseFloat(seg);
261        }
262    };
