002package com.asoroka.sidora.csvmetadata;
003
004import static com.google.common.collect.Iterators.advance;
005import static com.google.common.collect.Iterators.cycle;
006import static com.google.common.collect.Iterators.peekingIterator;
007import static com.google.common.collect.Lists.newArrayListWithCapacity;
008import static org.slf4j.LoggerFactory.getLogger;
009
010import java.util.Iterator;
011import java.util.List;
012
013import org.apache.commons.csv.CSVParser;
014import org.apache.commons.csv.CSVRecord;
015import org.slf4j.Logger;
016
017import com.asoroka.sidora.csvmetadata.heuristics.DataTypeHeuristic;
018import com.google.common.collect.AbstractIterator;
019import com.google.common.collect.PeekingIterator;
020
021/**
022 * Value scanning workflow. Handed a {@link CSVParser}, this class will scan through it and supply the values of
023 * fields to a "row" of {@link DataTypeHeuristic} strategies cloned from the configured choice.
024 * 
025 * @author ajs6f
026 */
027public class CsvScanner extends AbstractIterator<CSVRecord> {
028
029    private final PeekingIterator<CSVRecord> internalScanner;
030
031    private final Iterator<DataTypeHeuristic<?>> strategies;
032
033    private final List<DataTypeHeuristic<?>> rowOfStrategies;
034
035    private static final Logger log = getLogger(CsvScanner.class);
036
037    public CsvScanner(final CSVParser parser, final DataTypeHeuristic<?> strategy) {
038        super();
039        this.internalScanner = peekingIterator(parser.iterator());
040        final int numColumns = internalScanner.peek().size();
041
042        log.debug("Found {} columns in our CSV.", numColumns);
043        // create a "row" of strategy instances of the same length as the rows in our CSV
044        this.rowOfStrategies = newArrayListWithCapacity(numColumns);
045        for (int i = 0; i < numColumns; i++) {
046            rowOfStrategies.add(strategy.clone());
047        }
048        // this.strategies will cycle endlessly around our row
049        this.strategies = cycle(rowOfStrategies);
050    }
051
052    @Override
053    protected CSVRecord computeNext() {
054        if (internalScanner.hasNext()) {
055            final CSVRecord nextRecord = internalScanner.next();
056            for (final String value : nextRecord) {
057                strategies.next().addValue(value);
058            }
059            return nextRecord;
060        }
061        return endOfData();
062    }
063
064    /**
065     * Scan rows in our CSV up to a limit, exhausting values from the internal parser as we do. <br/>
066     * WARNING: Be careful about calling this more than once on a {@link CsvScanner}. The internal parser of a scanner
067     * cannot be reset.
068     * 
069     * @param limit The number of rows to scan, 0 for all rows.
070     */
071    public void scan(final int limit) {
072        if (limit == 0) {
073            while (hasNext()) {
074                next();
075            }
076        } else {
077            advance(this, limit);
078        }
079    }
080
081    /**
082     * Use this to recover and interrogate the strategy instances used in scanning.
083     * 
084     * @return the row of strategies used to determine the types of fields
085     */
086    public List<DataTypeHeuristic<?>> getStrategies() {
087        return rowOfStrategies;
088    }
089
090}
