002package com.asoroka.sidora.csvmetadata.heuristics;
003
004import static org.junit.Assert.assertEquals;
005import static org.slf4j.LoggerFactory.getLogger;
006
007import org.junit.Test;
008import org.slf4j.Logger;
009
010import com.asoroka.sidora.csvmetadata.datatype.DataType;
011import com.asoroka.sidora.csvmetadata.heuristics.FractionHeuristic;
012
013public class FractionHeuristicTest extends CountAggregatingHeuristicTestFrame<FractionHeuristic> {
014
015    private static final Logger log = getLogger(FractionHeuristicTest.class);
016
017    @Test
018    public void testActionWithGoodValues() {
019
020        log.trace("testActionWithGoodValues()...");
021        for (final DataType testType : DataType.values()) {
022            testHeuristic = newTestInstance();
023            for (final String testValue : goodValues.get(testType)) {
024                testHeuristic.addValue(testValue);
025            }
026            assertEquals("Didn't get the correct type for datatype " + testType + "!", testType, testHeuristic
027                    .mostLikelyType());
028        }
029    }
030
031    @Test
032    public void testActionWithOneBadValue() {
033        log.trace("testActionWithOneBadValue()...");
034        for (final DataType testType : oneBadValue.keySet()) {
035            testHeuristic = newTestInstance();
036            for (final String testValue : oneBadValue.get(testType)) {
037                testHeuristic.addValue(testValue);
038            }
039            assertEquals("Failed to admit datatype but should have!", testType, testHeuristic.mostLikelyType());
040        }
041    }
042
043    @Override
044    protected FractionHeuristic newTestInstance() {
045        return new FractionHeuristic(0.2);
046    }
047
048}
