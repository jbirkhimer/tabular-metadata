029/**
030 * Utilities for testing.
031 * 
032 * @author ajs6f
033 */
034public abstract class TestUtilities {
035
036    public static void addValues(final Heuristic<?, ?> strategy, final Iterable<?> values) {
037        for (final Object value : values) {
038            strategy.addValue(value.toString());
039        }
040    }
041
042    /**
043     * Extracts the most likely type selection from a {@link ValueHeuristic}
044     */
045    protected static final Function<TypeDeterminingHeuristic<?>, DataType> getMostLikelyType =
046            new Function<TypeDeterminingHeuristic<?>, DataType>() {
047
048                @Override
049                public DataType apply(final TypeDeterminingHeuristic<?> heuristic) {
050                    return heuristic.results().first();
051                }
052            };
053
054    final static <From> Callable1<From, PotentialAssignment> toPotentialAssignment() {
055        return new Callable1<From, PotentialAssignment>() {
056
057            @Override
058            public PotentialAssignment call(final From from) {
059                return PotentialAssignment.forValue(null, from);
060            }
061        };
062    }
063
064    /**
065     * @param type a {@link DataType}
066     * @return a random value appropriate to that DataType
067     */
068    static Comparable<?> generateRandomValue(final DataType type) {
069        switch (type) {
070        case Boolean:
071            return random() > 0.5;
072        case DateTime:
073            return now().plus(round(random() * 1000000000000F));
074        case Decimal:
075            return (float) (random() - 0.5) * 10;
076        case Geographic:
077            if ((Boolean) generateRandomValue(DataType.Boolean)) {
078                return new GeographicValue(asList(
079                        (Float) generateRandomValue(Decimal),
080                        (Float) generateRandomValue(Decimal)));
081            }
082            return new GeographicValue(asList(
083                    (Float) generateRandomValue(Decimal),
084                    (Float) generateRandomValue(Decimal),
085                    (Float) generateRandomValue(Decimal)));
086        case Integer:
087            if (random() < 0.1) {
088                if ((Boolean) generateRandomValue(Boolean)) {
089                    return MAX_VALUE;
090                }
091                return MIN_VALUE;
092            }
093            return round((Float) generateRandomValue(Decimal));
094        case NonNegativeInteger:
095            if (random() < 0.2) {
096                return 0;
097            }
098            final Integer randInt = (Integer) generateRandomValue(Integer);
099            if (randInt.equals(MAX_VALUE) || randInt.equals(MIN_VALUE)) {
100                return 0;
101            }
102            return abs(randInt);
103        case PositiveInteger:
104            return (Integer) generateRandomValue(NonNegativeInteger) + 1;
105        case String:
106            return randomUUID().toString();
107        case URI:
108            return URI.create("info:" + generateRandomValue(DataType.String));
109        default:
110            throw new AssertionError("A DataType of an un-enumerated kind should never exist!");
111        }
112    }
113
114    static RandomValuesForAType randomValues(final DataType type, final short numValues) {
115        final RandomValuesForAType values = new RandomValuesForAType(numValues).withType(type);
116        for (short valueIndex = 0; valueIndex < numValues; valueIndex++) {
117            values.add(generateRandomValue(type));
118        }
119        return values;
120    }
121
122    public static class RandomValuesForAType extends ArrayList<Comparable<?>> {
123
124        private static final long serialVersionUID = 1L;
125
126        public DataType type;
127
128        public RandomValuesForAType(final short initCapacity) {
129            super(initCapacity);
130        }
131
132        public RandomValuesForAType withType(final DataType t) {
133            this.type = t;
134            return this;
135        }
136    }
