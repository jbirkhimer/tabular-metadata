066     * @return a mock strategy

068    public static MockedHeuristic cloneableMockStrategy(final MockedHeuristic strategy) {
069        final MockedHeuristic mocked = mock(MockedHeuristic.class);
070        final Returns cloner = new Returns(strategy);
071        when(mocked.newInstance()).thenAnswer(cloner);
072        return mocked;
073    }

096    /**
097     * @param type a {@link DataType}
098     * @return a random value appropriate to that DataType
099     */
100    static Comparable<?> generateRandomValue(final DataType type) {
101        switch (type) {
102        case Boolean:
103            return random() > 0.5;
104        case DateTime:
105            return now().plus(round(random() * 1000000000000F));
106        case Decimal:
107            return (float) (random() - 0.5) * 10;
108        case Geographic:
109            if ((Boolean) generateRandomValue(DataType.Boolean)) {
110                return new GeographicValue(asList(
111                        (Float) generateRandomValue(Decimal),
112                        (Float) generateRandomValue(Decimal)));
113            }
114            return new GeographicValue(asList(
115                    (Float) generateRandomValue(Decimal),
116                    (Float) generateRandomValue(Decimal),
117                    (Float) generateRandomValue(Decimal)));
118        case Integer:
119            if (random() < 0.1) {
120                if ((Boolean) generateRandomValue(Boolean)) {
121                    return MAX_VALUE;
122                }
123                return MIN_VALUE;
124            }
125            return round((Float) generateRandomValue(Decimal));
126        case NonNegativeInteger:
127            if (random() < 0.2) {
128                return 0;
129            }
130            final Integer randInt = (Integer) generateRandomValue(Integer);
131            if (randInt.equals(MAX_VALUE) || randInt.equals(MIN_VALUE)) {
132                return 0;
133            }
134            return abs(randInt);
135        case PositiveInteger:
136            return (Integer) generateRandomValue(NonNegativeInteger) + 1;
137        case String:
138            return randomUUID().toString();
139        case URI:
140            return URI.create("info:" + generateRandomValue(DataType.String));
141        default:
142            throw new AssertionError("A DataType of an un-enumerated kind should never exist!");
143        }
144    }

146    static RandomValuesForAType randomValues(final DataType type, final short numValues) {
147        final RandomValuesForAType values = new RandomValuesForAType(numValues).withType(type);
148        for (short valueIndex = 0; valueIndex < numValues; valueIndex++) {
149            values.add(generateRandomValue(type));
150        }
151        return values;
152    }

154    public static class RandomValuesForAType extends ArrayList<Comparable<?>> {
155
156        private static final long serialVersionUID = 1L;
157
158        public DataType type;
159
160        public RandomValuesForAType(final short initCapacity) {
161            super(initCapacity);
162        }
163
164        public RandomValuesForAType withType(final DataType t) {
165            this.type = t;
166            return this;
167        }
168
169        public <T> List<T> as() {
170            @SuppressWarnings("unchecked")
171            final List<T> list = (List<T>) this;
172            return list;
173        }
174    }
