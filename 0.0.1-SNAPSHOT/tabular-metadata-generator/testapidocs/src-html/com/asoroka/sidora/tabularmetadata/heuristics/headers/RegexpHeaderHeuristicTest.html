002package com.asoroka.sidora.tabularmetadata.heuristics.headers;
003
004import static java.util.Arrays.asList;
005import static org.junit.Assert.assertFalse;
006import static org.junit.Assert.assertTrue;
007
008import java.util.List;
009
010import org.junit.Test;
011
012public class RegexpHeaderHeuristicTest extends HeaderHeuristicTestFrame<RegexpHeaderHeuristic> {
013
014    private static final String pattern = "MARKER";
015
016    private static final List<String> headerRow1 = asList("MARKER1", "MARKER2", "MARKER3");
017
018    private static final List<String> headerRow2 = asList("jdfhg", "saf", "324");
019
020    @Override
021    protected RegexpHeaderHeuristic newTestHeuristic() {
022        return new RegexpHeaderHeuristic(pattern);
023    }
024
025    @Test
026    public void test() {
027        final TreatsEachFieldAlikeHeaderHeuristic<RegexpHeaderHeuristic> testHeuristic = newTestHeuristic();
028        for (final String field : headerRow1) {
029            testHeuristic.addValue(field);
030        }
031        assertTrue("Failed to pass a row that should have passed!", testHeuristic.isHeader());
032        for (final String field : headerRow2) {
033            testHeuristic.addValue(field);
034        }
035        assertFalse("Passed a row that shouldn't have passed!", testHeuristic.isHeader());
036    }
037
038}
