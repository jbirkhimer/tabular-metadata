002package com.asoroka.sidora.tabularmetadata.heuristics.types;
003
004import static com.asoroka.sidora.tabularmetadata.datatype.DataType.String;
005import static com.asoroka.sidora.tabularmetadata.testframework.TestUtilities.addValues;
006import static java.util.UUID.randomUUID;
007import static org.hamcrest.CoreMatchers.is;
008import static org.hamcrest.core.IsNot.not;
009import static org.junit.Assert.assertEquals;
010import static org.junit.Assert.assertNotEquals;
011import static org.junit.Assume.assumeThat;
012import static org.slf4j.LoggerFactory.getLogger;
013
014import org.junit.experimental.theories.Theories;
015import org.junit.experimental.theories.Theory;
016import org.junit.runner.RunWith;
017import org.slf4j.Logger;
018
019import com.asoroka.sidora.tabularmetadata.testframework.RowsOfRandomValuesForAllTypes;
020import com.asoroka.sidora.tabularmetadata.testframework.TestUtilities.RandomValuesForAType;
021
022@RunWith(Theories.class)
023public class StrictHeuristicTest extends PerTypeHeuristicTestFrame<StrictHeuristic> {
024
025    private static final Logger log = getLogger(StrictHeuristicTest.class);
026
027    @Override
028    protected StrictHeuristic newTestHeuristic() {
029        return new StrictHeuristic();
030    }
031
032    @Theory
033    public void inputsWithOneUnparseableValueShouldNotBeRecognizedAsTheirTrueType(
034            @RowsOfRandomValuesForAllTypes(numRowsPerType = 5, valuesPerType = 50) final RandomValuesForAType values) {
035        // nothing cannot be recognized as a String
036        assumeThat(values.type, not(is(String)));
037        // but a UUID could only be recognized as a String
038        values.add(randomUUID());
039        final StrictHeuristic testHeuristic = newTestHeuristic();
040        addValues(testHeuristic, values);
041        assertNotEquals(values.type, testHeuristic.results().first());
042    }
043
044    @Theory
045    public void inputsWithNoUnparseableValuesShouldBeRecognizedAsTheirTrueType(
046            @RowsOfRandomValuesForAllTypes(numRowsPerType = 5, valuesPerType = 50) final RandomValuesForAType values) {
047        log.trace("StrictHeuristicTest chacking {}", values.type);
048        final StrictHeuristic testHeuristic = newTestHeuristic();
049        addValues(testHeuristic, values);
050        assertEquals(values.type, testHeuristic.results().first());
051    }
052}
