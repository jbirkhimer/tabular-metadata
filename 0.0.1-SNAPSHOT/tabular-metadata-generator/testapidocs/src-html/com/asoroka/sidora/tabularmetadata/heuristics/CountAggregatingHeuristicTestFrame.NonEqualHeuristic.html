030
031    protected static Map<DataType, List<String>> parseableValues;
032
033    protected static Map<DataType, List<String>> oneNonparseableValue;
034    {
035        {
036            // these are all parseable values for the datatype to which they are assigned
037            parseableValues = new EnumMap<>(DataType.class);
038            parseableValues.put(String, of("Jane", "John", "Sarah", "Simon"));
039            parseableValues.put(Decimal, of("0", "0.1", "-1", "34.345"));
040            parseableValues.put(Integer, of("0", "1", "-1 ", "34"));
041            parseableValues.put(NonNegativeInteger, of("0", "1", " 11", "34"));
042            parseableValues.put(PositiveInteger, of("354455", "13452432", "112345235 ", "34529534"));
043            parseableValues.put(Boolean, of("True", "F", "TruE", "FaLse"));
044            parseableValues.put(Geographic, of("38.03,-78.478889", " -78.478889,38.03", "1,0,77", "0,1"));
045            parseableValues.put(DateTime, of("1990-3-4", "2014-273", "2014-W40-2", "2014-W40",
046                    "2014-09-30T18:58:45Z"));
047
048            // copy the parseable values for reuse
049            oneNonparseableValue = new EnumMap<>(DataType.class);
050            for (final DataType type : DataType.values()) {
051                oneNonparseableValue.put(type, newArrayList(parseableValues.get(type)));
052
053            }
054            // here we add one nonparseable value to each list of parseable values
055            oneNonparseableValue.get(Decimal).add("0sd");
056            oneNonparseableValue.get(Integer).add("-1.3 ");
057            oneNonparseableValue.get(NonNegativeInteger).add("-11");
058            oneNonparseableValue.get(PositiveInteger).add("Q");
059            oneNonparseableValue.get(Boolean).add("Q");
060            oneNonparseableValue.get(Geographic).add("38.03");
061            oneNonparseableValue.get(DateTime).add("2014/24");
062            // nothing cannot parse as a String
063            oneNonparseableValue.remove(String);
064        }
065    }
066
067    @Test
068    public void testClone() {
069        testHeuristic = newTestInstance();
070        assertTrue(testHeuristic.clone().equals(testHeuristic));
071    }
072
073    @Test
074    public void testEquals() {
075        testHeuristic = newTestInstance();
076        assertTrue(testHeuristic.equals(newTestInstance()));
077        assertFalse(testHeuristic.equals(new Object()));
078
079        final CountAggregatingHeuristic<?> nonEqualObject = new NonEqualHeuristic();
080        nonEqualObject.addValue("test value");
081        assertFalse(testHeuristic.equals(nonEqualObject));
082    }

099        }
100    }
