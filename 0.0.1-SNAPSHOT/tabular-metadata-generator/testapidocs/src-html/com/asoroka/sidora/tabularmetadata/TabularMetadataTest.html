002package com.asoroka.sidora.tabularmetadata;
003
004import static org.junit.Assert.assertEquals;
005
006import java.util.List;
007import java.util.Map;
008import java.util.Set;
009import java.util.SortedSet;
010
011import org.junit.Test;
012import org.junit.runner.RunWith;
013import org.mockito.Mock;
014import org.mockito.runners.MockitoJUnitRunner;
015
016import com.asoroka.sidora.tabularmetadata.datatype.DataType;
017import com.google.common.collect.Range;
018
019@RunWith(MockitoJUnitRunner.class)
020public class TabularMetadataTest {
021
022    @Mock
023    private List<String> mockHeaderNames;
024
025    @Mock
026    private List<SortedSet<DataType>> mockTypes;
027
028    @Mock
029    private List<Map<DataType, Range<?>>> mockRanges;
030
031    @Mock
032    private List<Map<DataType, Set<String>>> mockEnumValues;
033
034    @Test
035    public void testContainerAction() {
036        final TabularMetadata testMetadata =
037                new TabularMetadata(mockHeaderNames, mockTypes, mockRanges, mockEnumValues);
038        assertEquals(mockHeaderNames, testMetadata.headerNames());
039        assertEquals(mockTypes, testMetadata.fieldTypes());
040        assertEquals(mockRanges, testMetadata.minMaxes());
041        assertEquals(mockEnumValues, testMetadata.enumeratedValues());
042    }
043
044}
