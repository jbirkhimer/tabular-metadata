020import com.asoroka.sidora.tabularmetadata.datatype.DataType;
021import com.asoroka.sidora.tabularmetadata.datatype.GeographicValue;

026/**
027 * Because of the enormous mix of data types in these tests, and because this is in fact test code, and most of all,
028 * because of Java's really pitiful lack of type variance, it is simpler here to dispense with some type safety, hence
029 * the annotation to this class.
030 * 
031 * @author ajs6f
032 */
033@SuppressWarnings("rawtypes")
034public class RunningMinMaxHeuristicTest extends HeuristicTestFrame<RunningMinMaxHeuristic> {
035
036    private static Map<DataType, Map<Range, List>> sampleData;
037
038    static {
039        sampleData =
040                ImmutableMap.<DataType, Map<Range, List>> builder()
041                        .put(
042                                DataType.Integer,
043                                of((Range) closed(-10, 10),
044                                        (List) newArrayList(-10, -5, 0, 5, 10)))
045                        .put(
046                                DataType.Decimal,
047                                of((Range) closed(-10F, 10F),
048                                        (List) newArrayList(-10F, -5F, 0F, 5F, 10F)))
049                        .put(
050                                DataType.String,
051                                of((Range) closed("BAR", "QUUX"),
052                                        (List) newArrayList("FOO", "BAR", "QUUX")))
053                        .put(
054                                DataType.Geographic,
055                                of((Range) closed(new GeographicValue(asList(0F, 0F)), new GeographicValue(asList(1F,
056                                        1F))),
057                                        (List) newArrayList("0,0", "0,1", "0,2", "1,0", "1,1")))
058                        .put(
059                                DataType.PositiveInteger,
060                                of((Range) closed(1, 10),
061                                        (List) newArrayList("1", "5", "10")))
062                        .put(
063                                DataType.NonNegativeInteger,
064                                of((Range) closed(0, 10),
065                                        (List) newArrayList("0", "1", "5", "10")))
066                        .put(
067                                DataType.URI,
068                                of((Range) closed(URI.create("http://a"), URI.create("http://z")),
069                                        (List) newArrayList("http://a", "http://k", "http://r", "http://z")))
070                        .put(
071                                DataType.Boolean,
072                                of(
073                                        (Range) closed(true, true), (List) newArrayList("true", "true", "true"),
074                                        (Range) closed(false, true), (List) newArrayList("true", "false", "true")))
075                        .put(
076                                DataType.DateTime,
077                                of((Range) closed(date("0"), date("100")),
078                                        (List) newArrayList("0", "25", "50", "100")))
079                        .build();
080
081    }
082
083    protected static DateTime date(final String v) {
084        return dateTimeParser().parseDateTime(v);
085    }
086
087    @Override
088    protected RunningMinMaxHeuristic newTestHeuristic() {
089        return new RunningMinMaxHeuristic();
090    }
091
092    private static final Logger log = getLogger(RunningMinMaxHeuristicTest.class);
093
094    @Test
095    public void testMinMaxsPerDataType() {
096        log.trace("testMinMaxsPerDataType()...");
097        for (final DataType testType : sampleData.keySet()) {
098            log.trace("Testing min/max for type: {}", testType);
099            final Map<Range, List> sampleRangeAndData = sampleData.get(testType);
100            final RunningMinMaxHeuristic testStrategy = newTestHeuristic();
101            for (final Range expectedRange : sampleRangeAndData.keySet()) {
102                log.trace("Found expected range: {}", expectedRange);
103                final List sampleDataForRange = sampleRangeAndData.get(expectedRange);
104                for (final Object lex : sampleDataForRange) {
105                    log.trace("Adding lex: {}", lex);
106                    testStrategy.addValue(lex.toString());
107                }
108                final Map<DataType, Range<?>> minMaxes = testStrategy.getRanges();
109                final Range minMax = minMaxes.get(testType);
110                assertEquals("Got wrong min/max!", expectedRange, minMax);
111            }
112        }
113    }
114
115    @Test
116    public void testMissingLimits() {
117        final RunningMinMaxHeuristic testStrategy = newTestHeuristic();
118        for (int i = -10; i <= -1; i++) {
119            testStrategy.addValue(String.valueOf(i));
120        }
121        // now we ask for a range for a data type for which it was not set
122        final Range<GeographicValue> geographicRange =
123                (Range<GeographicValue>) testStrategy.getRanges().get(DataType.Geographic);
124        assertEquals("Found a defined range where we should not have!", Range.all(), geographicRange);
125    }
126}
