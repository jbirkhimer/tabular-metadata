031
032    protected static Map<DataType, List<String>> oneBadValue;
033    {
034        {
035            // these are all good (parseable) values for the datatype to which they are assigned
036            goodValues = new EnumMap<>(DataType.class);
037            goodValues.put(String, of("Jane", "John", "Sarah", "Simon"));
038            goodValues.put(Decimal, of("0", "0.1", "-1", "34.345"));
039            goodValues.put(Integer, of("0", "1", "-1 ", "34"));
040            goodValues.put(NonNegativeInteger, of("0", "1", " 11", "34"));
041            goodValues.put(PositiveInteger, of("354455", "13452432", "112345235 ", "34529534"));
042            goodValues.put(Boolean, of("True", "F", "TruE", "FaLse"));
043            goodValues.put(Geographic, of("38.03,-78.478889", " -78.478889,38.03", "1,0,77", "0,1"));
044            goodValues.put(DateTime, of("1990-3-4", "2014-273", "2014-W40-2", "2014-W40",
045                    "2014-09-30T18:58:45Z"));
046
047            // copy the good values for reuse
048            oneBadValue = new EnumMap<>(DataType.class);
049            for (final DataType type : DataType.values()) {
050                oneBadValue.put(type, newArrayList(goodValues.get(type)));
051
052            }
053            // here we add one bad value to each list of good values
054            oneBadValue.get(Decimal).add("0sd");
055            oneBadValue.get(Integer).add("-1.3 ");
056            oneBadValue.get(NonNegativeInteger).add("-11");
057            oneBadValue.get(PositiveInteger).add("Q");
058            oneBadValue.get(Boolean).add("Q");
059            oneBadValue.get(Geographic).add("38.03");
060            oneBadValue.get(DateTime).add("2014/24");
061            // nothing cannot parse as a String
062            oneBadValue.remove(String);
063        }
064    }
