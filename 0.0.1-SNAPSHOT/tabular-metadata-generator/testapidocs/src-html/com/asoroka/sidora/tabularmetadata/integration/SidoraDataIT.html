015import java.io.File;
016import java.io.IOException;
017import java.net.MalformedURLException;
018import java.net.URL;
019import java.util.List;
020import java.util.SortedSet;
021import java.util.regex.Pattern;
022
023import org.junit.Test;
024import org.slf4j.Logger;
025
026import com.asoroka.sidora.tabularmetadata.TabularMetadata;
027import com.asoroka.sidora.tabularmetadata.TabularMetadataGenerator;
028import com.asoroka.sidora.tabularmetadata.datatype.DataType;
029import com.asoroka.sidora.tabularmetadata.testframework.TestUtilities;
030
031public class SidoraDataIT {
032
033    private static final Pattern DEFAULT_HEADER_NAME = compile("^Variable");
034
035    private static final String ITEST_DATA_DIR = "src/test/resources/itest-data";
036
037    private static final URL testSIFile1, testSIFile2;
038
039    private static final List<URL> testFiles;
040    static {
041        final File testFileDir = new File(ITEST_DATA_DIR);
042        try {
043            testSIFile1 = new File(testFileDir, "Thompson-WMA-10B-researcher_observation.csv").toURI().toURL();
044            testSIFile2 = new File(testFileDir, "Thompson-WMA-16C-researcher_observation.csv").toURI().toURL();
045            testFiles = asList(testSIFile1, testSIFile2);
046        } catch (final MalformedURLException e) {
047            throw new AssertionError("Couldn't find test files!");
048        }
049
050    }
051
052    private static final Logger log = getLogger(SidoraDataIT.class);
053
054    private static final List<DataType> expectedTypes =
055            asList(DataType.String, DataType.String, DataType.String, DateTime,
056                    DateTime, DataType.String, DataType.String, DataType.String, DataType.String,
057                    DataType.String, PositiveInteger, DataType.String, DataType.String, DataType.String);
058
059    @Test
060    public void testSIfiles() throws IOException {
061        for (final URL testFile : testFiles) {
062            testFile(testFile);
063        }
064    }
065
066    private static void testFile(final URL testFile) throws IOException {
067        final TabularMetadataGenerator testGenerator = new TabularMetadataGenerator();
068        final TabularMetadata result = testGenerator.getMetadata(testFile);
069        log.debug("Got results: {}", result);
070        assertTrue("Should have found all header names matching against '" + DEFAULT_HEADER_NAME + "'!",
071                all(result.headerNames(), contains(DEFAULT_HEADER_NAME)));
072
073        final List<DataType> mostLikelyTypes = getFirstElements(result.fieldTypes());
074        assertEquals("Didn't get the expected type determinations!", expectedTypes, mostLikelyTypes);
075
076        for (int i = 0; i < result.minMaxes().size(); i++) {
077            final DataType mostLikelyType = mostLikelyTypes.get(i);
078            log.debug("For most likely type {} got range: {}", mostLikelyType, result.minMaxes().get(i).get(
079                    mostLikelyType));
080        }
081    }
082
083    private static <T> List<T> getFirstElements(final List<SortedSet<T>> inputs) {
084        return sequence(inputs).map(TestUtilities.<T> first()).toList();
085    }
086}
