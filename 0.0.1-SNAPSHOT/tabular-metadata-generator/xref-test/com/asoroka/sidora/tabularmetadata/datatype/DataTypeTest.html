122     @Test
123     public void testOrderingByHierarchy() {
124         List<DataType> listToBeSorted = asList(Decimal, PositiveInteger, Integer, NonNegativeInteger, String);
125         List<DataType> listInCorrectSorting = asList(PositiveInteger, NonNegativeInteger, Integer, Decimal, String);
126         List<DataType> sorted = new ArrayList<>(sortByHierarchy(listToBeSorted));
127         assertEquals("Got wrong ordering by hierarchy of datatypes!", listInCorrectSorting, sorted);
128 
129         listToBeSorted = asList(String, Geographic);
130         listInCorrectSorting = asList(Geographic, String);
131         sorted = new ArrayList<>(sortByHierarchy(listToBeSorted));
132         assertEquals("Got wrong ordering by hierarchy of datatypes!", listInCorrectSorting, sorted);
133 
134         listToBeSorted = asList(String, DateTime);
135         listInCorrectSorting = asList(DateTime, String);
136         sorted = new ArrayList<>(sortByHierarchy(listToBeSorted));
137         assertEquals("Got wrong ordering by hierarchy of datatypes!", listInCorrectSorting, sorted);
138 
139         listToBeSorted = asList(String, Boolean);
