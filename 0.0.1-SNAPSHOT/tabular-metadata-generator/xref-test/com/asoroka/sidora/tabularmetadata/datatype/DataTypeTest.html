109     @Test
110     public void testCorrectParsingOfValues() {
111         for (final DataType testType : DataType.values()) {
112             for (final String testValue : sampleValues.get(testType)) {
113                 final Set<DataType> result = parseableAs(testValue);
114                 assertEquals("Did not find the appropriate datatypes suggested as parseable for a " + testType + "!", expectedParseableTypes
115                         .get(testType), result);
116             }
117         }
118     }

135     @Test
136     public void testOrderingByHierarchy() {
137         List<DataType> listToBeSorted = asList(Decimal, PositiveInteger, Integer, NonNegativeInteger, String);
138         List<DataType> listInCorrectSorting = asList(PositiveInteger, NonNegativeInteger, Integer, Decimal, String);
139         List<DataType> sorted = new ArrayList<>(sortByHierarchy(listToBeSorted));
140         assertEquals("Got wrong ordering by hierarchy of datatypes!", listInCorrectSorting, sorted);
141 
142         listToBeSorted = asList(String, Geographic);
143         listInCorrectSorting = asList(Geographic, String);
144         sorted = new ArrayList<>(sortByHierarchy(listToBeSorted));
145         assertEquals("Got wrong ordering by hierarchy of datatypes!", listInCorrectSorting, sorted);
146 
147         listToBeSorted = asList(String, DateTime);
148         listInCorrectSorting = asList(DateTime, String);
149         sorted = new ArrayList<>(sortByHierarchy(listToBeSorted));
150         assertEquals("Got wrong ordering by hierarchy of datatypes!", listInCorrectSorting, sorted);
151 
152         listToBeSorted = asList(String, Boolean);
153         listInCorrectSorting = asList(Boolean, String);
154         sorted = new ArrayList<>(sortByHierarchy(listToBeSorted));
155         assertEquals("Got wrong ordering by hierarchy of datatypes!", listInCorrectSorting, sorted);
156 
157         listToBeSorted = asList(String, URI);
158         listInCorrectSorting = asList(URI, String);
159         sorted = new ArrayList<>(sortByHierarchy(listToBeSorted));
160         assertEquals("Got wrong ordering by hierarchy of datatypes!", listInCorrectSorting, sorted);
161     }

163     @Test
164     public void testBadGeographies() {
165         String testValue = "23, 23, 23, 23";
166         assertFalse("Accepted a four-valued tuple as geographic coordinates!", parseableAs(testValue)
167                 .contains(Geographic));
168         testValue = "23";
169         assertFalse("Accepted a one-valued tuple as geographic coordinates!", parseableAs(testValue)
170                 .contains(Geographic));
171     }
