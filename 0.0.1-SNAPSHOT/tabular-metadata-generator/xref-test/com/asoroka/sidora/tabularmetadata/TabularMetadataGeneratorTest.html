138     @Test
139     public void testOperationWithSetFormat() throws IOException {
140         log.trace("Entering testOperationWithSetFormat()...");
141         when(mockHeaderHeuristic.results()).thenReturn(true);
142         final URL mockURL = mockURL(testTsv);
143         final TabularMetadataGenerator testParser =
144                 newGenerator(mockRangeStrategy, mockEnumStrategy, mockTypeStrategy);
145         testParser.setFormat(new TabularFormat.TabSeparated());
146         testParser.setHeaderStrategy(mockHeaderHeuristic);
147         final TabularMetadata results = testParser.getMetadata(mockURL);
148 
149         final List<String> headers = results.headerNames();
150         final List<NavigableMap<DataType, Range<?>>> ranges = results.minMaxes();
151         assertEquals(asList(testHeaders.split(",")), headers);
152         final List<SortedSet<DataType>> types = results.fieldTypes();
153         assertTrue("Found a data type that didn't originate from our mocking!", all(concat(types),
154                 equalTo(mockDataType)));
155         final boolean rangeTest = all(ranges, equalTo(testRanges()));
156         assertTrue(rangeTest);
157     }

159     @Test
160     public void testOperationWithHeaders() throws IOException {
161         log.trace("Entering testOperationWithHeaders()...");
162         when(mockHeaderHeuristic.results()).thenReturn(true);
163         final URL mockURL = mockURL(testCsv);
164         final TabularMetadataGenerator testParser =
165                 newGenerator(mockRangeStrategy, mockEnumStrategy, mockTypeStrategy);

168 
169         final List<SortedSet<DataType>> types = results.fieldTypes();
170         assertTrue(all(concat(types), equalTo(mockDataType)));
171 
172         final List<NavigableMap<DataType, Range<?>>> ranges = results.minMaxes();
173         assertTrue("Failed to find the right mock range results!", all(ranges, equalTo(testRanges())));
174         final List<String> headers = results.headerNames();
175         assertEquals("Got a bad list of header names!", asList(testHeaders.split(",")), headers);
176 
177         for (final Map<DataType, Set<String>> v : results.enumeratedValues()) {
178             assertEquals("Got a bad value as part of the enumerated list!", mockEnumeratedValues, v);
179         }

206     @Test
207     public void testOperationWithoutHeadersWithScanLimit() throws IOException {
208         log.trace("Entering testOperationWithoutHeadersWithScanLimit()...");
209         when(mockHeaderHeuristic.results()).thenReturn(false);
210         final URL mockURL = mockURL(testCsvWithMarker);
211         final MarkingMockDataTypeHeuristic testStrategy = new MarkingMockDataTypeHeuristic(MARKER_VALUE);
212         final TabularMetadataGenerator testParser =
213                 newGenerator(mockRangeStrategy, mockEnumStrategy, testStrategy);
214         testParser.setHeaderStrategy(mockHeaderHeuristic);
215         testParser.setScanLimit(2);
216         testParser.getMetadata(mockURL);
217         assertFalse("Discovered the marker in a row we should not have been scanning!",
218                 testStrategy.hasMarkerBeenSeen);
219         log.trace("Did not discover the marker in a row we should not have been scanning.");
220         testParser.setScanLimit(3);
221         testParser.getMetadata(mockURL);
222         assertTrue("Failed to discover the marker in a row we should have been scanning!",
223                 testStrategy.hasMarkerBeenSeen);
224         log.trace("Discovered the marker in a row we should have been scanning.");
225         log.trace("Done with testOperationWithoutHeadersWithScanLimit().");
226     }

251     private static class MarkingMockDataTypeHeuristic implements
252             TypeDeterminingHeuristic<MarkingMockDataTypeHeuristic> {
253 
254         private static final Logger markLog = getLogger(MarkingMockDataTypeHeuristic.class);
255 
256         /**
257          * @param marker
258          */
259         public MarkingMockDataTypeHeuristic(final String marker) {
260             this.marker = marker;
261         }
262 
263         private String marker;
264 
265         public boolean hasMarkerBeenSeen = false;
266 
267         @Override
268         public SortedSet<DataType> results() {
269             return sortByHierarchy(DataType.valuesSet());
270         }
271 
272         @Override
273         public MarkingMockDataTypeHeuristic newInstance() {
274             return this;
275         }
276 
277         @Override
278         public boolean addValue(final String lex) {
279             markLog.trace("Checking lex {} for marker {}", lex, marker);
280             if (lex.equals(marker)) {
281                 hasMarkerBeenSeen = true;
282             }
283             return true;
284         }
285 
286         @Override
287         public MarkingMockDataTypeHeuristic get() {
288             return this;
289         }
290 
291         @Override
292         public void reset() {
293             // NO OP
294         }
