66      private static void testFile(final URL testFile) throws IOException {
67          final TabularMetadataGenerator testGenerator = new TabularMetadataGenerator();
68          final TabularMetadata result = testGenerator.getMetadata(testFile);
69          log.debug("Got results: {}", result);
70          assertTrue("Should have found all header names matching against '" + DEFAULT_HEADER_NAME + "'!",
71                  all(result.headerNames(), contains(DEFAULT_HEADER_NAME)));
72  
73          final List<DataType> mostLikelyTypes = getFirstElements(result.fieldTypes());
74          assertEquals("Didn't get the expected type determinations!", expectedTypes, mostLikelyTypes);
75  
76          for (int i = 0; i < result.minMaxes().size(); i++) {
77              final DataType mostLikelyType = mostLikelyTypes.get(i);
78              log.debug("For most likely type {} got range: {}", mostLikelyType, result.minMaxes().get(i).get(
79                      mostLikelyType));
80          }
81      }
