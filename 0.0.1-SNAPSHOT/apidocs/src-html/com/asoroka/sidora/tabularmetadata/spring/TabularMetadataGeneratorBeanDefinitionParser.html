002package com.asoroka.sidora.tabularmetadata.spring;
003
004import static java.lang.Integer.parseInt;
005import static org.springframework.beans.factory.support.AbstractBeanDefinition.AUTOWIRE_BY_TYPE;
006import static org.springframework.util.StringUtils.hasText;
007
008import org.springframework.beans.factory.support.BeanDefinitionBuilder;
009import org.springframework.beans.factory.xml.AbstractSingleBeanDefinitionParser;
010import org.w3c.dom.Element;
011
012import com.asoroka.sidora.tabularmetadata.TabularMetadataGenerator;
013
014/**
015 * Responsible for constructing {@link TabularMetadataGenerator} beans.
016 * 
017 * @author ajs6f
018 */
019public class TabularMetadataGeneratorBeanDefinitionParser extends AbstractSingleBeanDefinitionParser {
020
021    @Override
022    protected Class<?> getBeanClass(final Element element) {
023        return TabularMetadataGenerator.class;
024    }
025
026    @Override
027    protected void doParse(final Element element, final BeanDefinitionBuilder builder) {
028        builder.setAutowireMode(AUTOWIRE_BY_TYPE);
029        final String scanLimit = element.getAttribute("scanLimit");
030        if (hasText(scanLimit)) {
031            builder.addPropertyValue("scanLimit", parseInt(scanLimit));
032        }
033        final String formatBean = element.getAttribute("format");
034        if (hasText(formatBean)) {
035            builder.addPropertyReference("format", formatBean);
036        }
037        final String strategyBean = element.getAttribute("strategy");
038        if (hasText(strategyBean)) {
039            builder.addPropertyReference("strategy", strategyBean);
040        }
041        final String headerStrategyBean = element.getAttribute("headerStrategy");
042        if (hasText(headerStrategyBean)) {
043            builder.addPropertyReference("headerStrategy", headerStrategyBean);
044        }
045    }
046
047}
