002package com.asoroka.sidora.tabularmetadata;
003
004import static com.googlecode.totallylazy.Sequences.sequence;
005
006import java.util.Comparator;
007import java.util.List;
008import java.util.Map;
009import java.util.NavigableMap;
010import java.util.Set;
011import java.util.SortedSet;
012import java.util.TreeMap;
013
014import com.asoroka.sidora.tabularmetadata.datatype.DataType;
015import com.google.common.base.MoreObjects;
016import com.google.common.base.MoreObjects.ToStringHelper;
017import com.google.common.collect.Range;
018import com.googlecode.totallylazy.Function1;
019import com.googlecode.totallylazy.Pair;
020
021/**
022 * A container for the results of metadata extraction on a single data file, with machinery to present those results
023 * in appropriate orders.
024 * 
025 * @author ajs6f
026 */
027public class TabularMetadata {
028
029    private final List<String> headerNames;
030
031    private final List<SortedSet<DataType>> fieldTypes;
032
033    private final List<Map<DataType, Range<?>>> minMaxes;
034
035    private final List<Map<DataType, Set<String>>> enumeratedValues;
036
037    /**
038     * @param headerNames
039     * @param fieldTypes
040     * @param minMaxes
041     * @param enumeratedValues
042     */
043    public TabularMetadata(final List<String> headerNames, final List<SortedSet<DataType>> fieldTypes,
044            final List<Map<DataType, Range<?>>> minMaxes, final List<Map<DataType, Set<String>>> enumeratedValues) {
045        this.headerNames = headerNames;
046        this.fieldTypes = fieldTypes;
047        this.minMaxes = minMaxes;
048        this.enumeratedValues = enumeratedValues;
049    }
050
051    /**
052     * Sorts a {@link Map<DataType,T>} in the same way that the corresponding field in {@link #fieldTypes()} is
053     * sorted. WARNING: Must not be used before setting {@link #fieldTypes}!
054     */
055    private <T> Function1<Pair<Number, Map<DataType, T>>, NavigableMap<DataType, T>> sortByLikelihood() {
056        return new Function1<Pair<Number, Map<DataType, T>>, NavigableMap<DataType, T>>() {
057
058            @Override
059            public NavigableMap<DataType, T> call(final Pair<Number, Map<DataType, T>> minMaxMapWithIndex) {
060                final Comparator<? super DataType> sortByLikelihood =
061                        fieldTypes().get(minMaxMapWithIndex.getKey().intValue()).comparator();
062                final TreeMap<DataType, T> sortedByLikelihood = new TreeMap<>(sortByLikelihood);
063                sortedByLikelihood.putAll(minMaxMapWithIndex.getValue());
064                return sortedByLikelihood;
065            }
066        };
067    }
068
069    /**
070     * @return A list of prospective header names.
071     */
072    public List<String> headerNames() {
073        return headerNames;
074    }
075
076    /**
077     * @return A list of candidates for field datatypes, one candidate set per field. Each candidate set is sorted
078     *         according to a notion of decreasing plausibility that is specific to the type-determination strategy
079     *         used.
080     */
081    public List<SortedSet<DataType>> fieldTypes() {
082        return fieldTypes;
083    }
084
085    /**
086     * @return A list of maps from {@link DataType}s to {@link Range}s, with each data type mapped to the minimum and
087     *         maximum for each field, <i>if</i> that field is treated as of that type. The maps are sorted in the
088     *         same manner as is {@link #fieldTypes()}. The values of the endpoints of the ranges, if they exist, are
089     *         in the Java type value-space associated to the datatype considered most likely for that field. The idea
090     *         here is that when a given field is finally determined to be of a given type (by user action after
091     *         automatic action), the appropriate range can be looked up at that time. This is to ensure that in a
092     *         situation where the type determination strategy employed gave a wrong answer, the correct answer for
093     *         range can still be found after the type determination has been corrected.
094     * @see com.google.common.collect.Range<?>
095     */
096    public List<NavigableMap<DataType, Range<?>>> minMaxes() {
097        return sequence(minMaxes).zipWithIndex().map(this.<Range<?>> sortByLikelihood()).toList();
098    }
099
100    /**
101     * @return A list (one element for each field in the input file) of maps from each possible datatype to an set of
102     *         the lexes found in that field that were parseable into that data type. The maps are sorted in the same
103     *         manner as is {@link #fieldTypes()}.
104     */
105    public List<NavigableMap<DataType, Set<String>>> enumeratedValues() {
106        return sequence(enumeratedValues).zipWithIndex().map(this.<Set<String>> sortByLikelihood()).toList();
107    }
108
109    private static final ToStringHelper toStringHelper() {
110        return MoreObjects.toStringHelper(TabularMetadata.class);
111    }
112
113    @Override
114    public String toString() {
115        return toStringHelper().add("headerNames", headerNames()).add("fieldTypes", fieldTypes())
116                .add("enumeratedValues", enumeratedValues()).add("minMaxes", minMaxes()).toString();
117    }
118}
