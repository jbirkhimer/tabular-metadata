002package com.asoroka.sidora.tabularmetadata.web;
003
004import static com.google.common.collect.Lists.transform;
005import static javax.xml.bind.annotation.XmlAccessType.PUBLIC_MEMBER;
006import static org.slf4j.LoggerFactory.getLogger;
007
008import java.util.List;
009import java.util.SortedMap;
010import java.util.SortedSet;
011
012import javax.xml.bind.annotation.XmlAccessorType;
013import javax.xml.bind.annotation.XmlElement;
014import javax.xml.bind.annotation.XmlElementWrapper;
015import javax.xml.bind.annotation.XmlRootElement;
016
017import org.slf4j.Logger;
018
019import com.asoroka.sidora.tabularmetadata.TabularMetadata;
020import com.asoroka.sidora.tabularmetadata.datatype.DataType;
021import com.google.common.base.Function;
022import com.google.common.collect.Range;
023
024/**
025 * A JAXB-supported class to serialize a precis of tabular data metadata.
026 * 
027 * @author ajs6f
028 */
029@XmlRootElement
030@XmlAccessorType(PUBLIC_MEMBER)
031public class TabularMetadataPrecis {
032
033    private TabularMetadata metadata;
034
035    private static final Logger log = getLogger(TabularMetadataPrecis.class);
036
037    /**
038     * Constructor for JAX-RS
039     */
040    public TabularMetadataPrecis() {
041    }
042
043    /**
044     * Real default constructor.
045     * 
046     * @param m
047     */
048    public TabularMetadataPrecis(final TabularMetadata m) {
049        this.metadata = m;
050    }
051
052    @XmlElementWrapper
053    @XmlElement(name = "mostLikelyHeader")
054    public List<String> getLikelyHeaders() {
055        return metadata.headerNames();
056    }
057
058    @XmlElementWrapper
059    @XmlElement(name = "mostLikelyType")
060    public List<DataType> getMostLikelyTypes() {
061        return transform(metadata.fieldTypes(), TabularMetadataPrecis.<DataType> first());
062    }
063
064    @XmlElementWrapper
065    @XmlElement(name = "mostLikelyRange")
066    public List<XMLRange> getMostLikelyRanges() {
067        return transform(metadata.minMaxes(), toXMLRange);
068    }
069
070    private static final <T> Function<SortedSet<T>, T> first() {
071        return new Function<SortedSet<T>, T>() {
072
073            @Override
074            public T apply(final SortedSet<T> s) {
075                return s.first();
076            }
077        };
078    }
079
080    private static final Function<SortedMap<DataType, Range<?>>, XMLRange> toXMLRange =
081            new Function<SortedMap<DataType, Range<?>>, XMLRange>() {
082
083                @Override
084                public XMLRange apply(final SortedMap<DataType, Range<?>> minMaxForField) {
085                    final Range<?> minMax = minMaxForField.get(minMaxForField.firstKey());
086                    return new XMLRange().range(minMax);
087                }
088            };
089
090    @XmlRootElement
091    @XmlAccessorType(PUBLIC_MEMBER)
092    private static class XMLRange {
093
094        private Range<?> range;
095
096        public XMLRange() {
097        }
098
099        public XMLRange range(final Range<?> r) {
100            this.range = r;
101            return this;
102        }
103
104        @XmlElement
105        public String getMin() {
106            return range.hasLowerBound() ? range.lowerEndpoint().toString() : null;
107        }
108
109        @XmlElement
110        public String getMax() {
111            return range.hasUpperBound() ? range.upperEndpoint().toString() : null;
112        }
113    }
114}
