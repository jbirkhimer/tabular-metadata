002package com.asoroka.sidora.tabularmetadata.heuristics.headers;
003
004import static com.google.common.base.Predicates.contains;
005import static java.util.regex.Pattern.compile;
006import static org.slf4j.LoggerFactory.getLogger;
007
008import java.util.regex.Pattern;
009
010import org.slf4j.Logger;
011
012import com.google.common.base.Predicate;
013
014/**
015 * A {@link HeaderHeuristic} that uses a regular expression applied to each field in a row to determine whether the
016 * row is a header row.
017 * 
018 * @author ajs6f
019 */
020public class RegexpHeaderHeuristic extends TreatsEachFieldAlikeHeaderHeuristic<RegexpHeaderHeuristic> {
021
022    private final Pattern regexp;
023
024    private static final Logger log = getLogger(RegexpHeaderHeuristic.class);
025
026    /**
027     * @param regexp The regular expression against which to match.
028     */
029    public RegexpHeaderHeuristic(final String regexp) {
030        log.debug("Using {} for header determination with pattern {}.", this.getClass(), regexp);
031        this.regexp = compile(regexp);
032    }
033
034    @Override
035    protected Predicate<CharSequence> fieldTest() {
036        return contains(regexp);
037    }
038
039    @Override
040    public RegexpHeaderHeuristic get() {
041        return this;
042    }
043
044    @Override
045    public RegexpHeaderHeuristic newInstance() {
046        return new RegexpHeaderHeuristic(regexp.pattern());
047    }
048}
