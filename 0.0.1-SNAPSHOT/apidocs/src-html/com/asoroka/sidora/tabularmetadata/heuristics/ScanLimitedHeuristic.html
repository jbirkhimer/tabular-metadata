002package com.asoroka.sidora.tabularmetadata.heuristics;
003
004import javax.inject.Inject;
005
006/**
007 * Operates wrapped around an inner heuristic, but guards it by flagging (via the return value of
008 * {@link #addValue(String)}) when lex evaluation should cease. This condition is reached when more lexes have been
009 * seen than {@link #scanLimit} allows.
010 * 
011 * @author ajs6f
012 * @param <InnerHeuristicType>
013 */
014public class ScanLimitedHeuristic<InnerHeuristicType extends Heuristic<InnerHeuristicType>>
015        extends ValueCountingHeuristic<ScanLimitedHeuristic<InnerHeuristicType>> {
016
017    private final InnerHeuristicType innerHeuristic;
018
019    private final int scanLimit;
020
021    /**
022     * @param innerHeuristic
023     * @param scanLimit
024     */
025    @Inject
026    public ScanLimitedHeuristic(final InnerHeuristicType innerHeuristic, final int scanLimit) {
027        super();
028        this.innerHeuristic = innerHeuristic;
029        this.scanLimit = scanLimit;
030    }
031
032    /**
033     * We test to see if we have passed the scan limit and only continue processing this lex if we have not.
034     * 
035     * @see com.asoroka.sidora.tabularmetadata.heuristics.ValueCountingHeuristic#addValue(java.lang.String)
036     */
037    @Override
038    public boolean addValue(final String lex) {
039        if (super.addValue(lex) && totalNumValues() <= scanLimit) {
040            return innerHeuristic.addValue(lex);
041        }
042        return false;
043    }
044
045    @Override
046    public void reset() {
047        super.reset();
048        innerHeuristic.reset();
049    }
050
051    @Override
052    public ScanLimitedHeuristic<InnerHeuristicType> get() {
053        return this;
054    }
055
056    @Override
057    public ScanLimitedHeuristic<InnerHeuristicType> clone() {
058        return new ScanLimitedHeuristic<>(innerHeuristic.clone(), scanLimit);
059    }
060
061}
