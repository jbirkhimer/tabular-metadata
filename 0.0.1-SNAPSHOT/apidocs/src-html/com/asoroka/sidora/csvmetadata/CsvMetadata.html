002package com.asoroka.sidora.csvmetadata;
003
004import java.util.List;
005
006import com.asoroka.sidora.csvmetadata.datatype.DataType;
007import com.google.common.collect.Range;
008
009/**
010 * @author ajs6f
011 */
012public class CsvMetadata {
013
014    private final List<String> headerNames;
015
016    private final List<DataType> columnTypes;
017
018    private final List<Range<?>> minMaxes;
019
020    /**
021     * @param headerNames
022     * @param columnTypes
023     * @param minMaxes
024     */
025    public CsvMetadata(final List<String> headerNames, final List<DataType> columnTypes,
026            final List<Range<?>> minMaxes) {
027        this.headerNames = headerNames;
028        this.columnTypes = columnTypes;
029        this.minMaxes = minMaxes;
030    }
031
032    public List<String> headerNames() {
033        return headerNames;
034    }
035
036    public List<DataType> columnTypes() {
037        return columnTypes;
038    }
039
040    public List<Range<?>> minMaxes() {
041        return minMaxes;
042    }
043}
