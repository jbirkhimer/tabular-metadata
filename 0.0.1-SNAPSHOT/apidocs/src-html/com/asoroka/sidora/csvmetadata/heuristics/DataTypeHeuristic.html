006package com.asoroka.sidora.csvmetadata.heuristics;
007
008import javax.inject.Provider;
009
010import com.asoroka.sidora.csvmetadata.datatype.DataType;
011import com.google.common.collect.Range;
012
013/**
014 * Determines into which {@link DataType} a series of values most likely falls. Also determines a minimum and maximum
015 * of the supplied values.
016 * 
017 * @author ajs6f
018 * @param <T>
019 */
020public interface DataTypeHeuristic<T extends DataTypeHeuristic<T>> extends Cloneable, Provider<DataTypeHeuristic<T>> {
021
022    /**
023     * @return The most likely type for the proffered values.
024     */
025    public DataType mostLikelyType();
026
027    /**
028     * Provide a value to this heuristic for consideration.
029     * 
030     * @param value the value to consider
031     */
032    public void addValue(final String value);
033
034    /**
035     * @return The range taken on by proffered values in the value space of the most likely type.
036     */
037    public Range<?> getRange();
038
039    /**
040     * We override {@link Object.clone()} in order to narrow its return type for type-safety.
041     * 
042     * @return A clone of this heuristic.
043     */
044    public abstract T clone();
045
046}
