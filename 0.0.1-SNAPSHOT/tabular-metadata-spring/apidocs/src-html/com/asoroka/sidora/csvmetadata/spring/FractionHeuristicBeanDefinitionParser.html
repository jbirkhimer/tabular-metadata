002package com.asoroka.sidora.csvmetadata.spring;
003
004import static java.lang.Float.parseFloat;
005
006import org.springframework.beans.factory.support.BeanDefinitionBuilder;
007import org.springframework.beans.factory.xml.AbstractSingleBeanDefinitionParser;
008import org.w3c.dom.Element;
009
010import com.asoroka.sidora.csvmetadata.heuristics.FractionHeuristic;
011
012/**
013 * Responsible for constructing {@link FractionHeuristic} beans.
014 * 
015 * @author ajs6f
016 */
017public class FractionHeuristicBeanDefinitionParser extends AbstractSingleBeanDefinitionParser {
018
019    @Override
020    protected Class<?> getBeanClass(final Element element) {
021        return FractionHeuristic.class;
022    }
023
024    @Override
025    protected void doParse(final Element element, final BeanDefinitionBuilder builder) {
026        final String fraction = element.getAttribute("fraction");
027        builder.addConstructorArgValue(parseFloat(fraction));
028    }
029}
