002package com.asoroka.sidora.tabularmetadata.spring;
003
004import static java.util.Arrays.asList;
005import static org.junit.Assert.assertEquals;
006import static org.slf4j.LoggerFactory.getLogger;
007
008import java.io.IOException;
009import java.net.MalformedURLException;
010import java.util.List;
011
012import org.junit.Test;
013import org.slf4j.Logger;
014import org.springframework.test.context.ContextConfiguration;
015
016import com.asoroka.sidora.tabularmetadata.TabularMetadata;
017import com.asoroka.sidora.tabularmetadata.datatype.DataType;
018
019@ContextConfiguration("/spring-xml/operation-with-specified-header-strategy.xml")
020public class OperationWithSpecifiedHeaderStrategyIT extends SpringITFramework {
021
022    private static String testFileSimpleWithCrazyHeaders = "simple-with-crazy-headers.csv";
023
024    private static final Logger log = getLogger(OperationWithSpecifiedHeaderStrategyIT.class);
025
026    @Test
027    public void testWithSimpleData() throws MalformedURLException, IOException {
028        log.trace("testWithSimpleData()...");
029        final TabularMetadata result =
030                testFile(getTestFile(testFileSimple), STRING_TYPES, getStringRange(), DataType.String);
031        final List<String> headerNames = result.headerNames();
032        final int numHeaders = headerNames.size();
033        for (int i = 0; i < numHeaders; i++) {
034            assertEquals("Found wrong name for header!", "Variable " + i, headerNames.get(i));
035
036        }
037        log.trace("End testWithSimpleData().");
038    }
039
040    @Test
041    public void testWithDataWithHeadersMatching() throws MalformedURLException, IOException {
042        log.trace("testWithDataWithHeadersMatching()...");
043        final TabularMetadata result =
044                testFile(getTestFile(testFileSimpleWithCrazyHeaders), SIMPLE_TYPES, getIntRange(), DataType.Integer);
045        assertEquals("Didn't find header names when we should have!", asList("MATCH1", "MATCH2", "MATCH3"), result
046                .headerNames());
047        log.trace("End testWithDataWithHeadersMatching().");
048    }
049}
