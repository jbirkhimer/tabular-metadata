46      private static final Charset CHARACTER_ENCODING = UTF_8;
47  
48      private CSVFormat format = DEFAULT;
49  
50      /**
51       * DefaultHeaderHeuristic value of {@code 0} indicates no limit. See {@link TabularScanner.scan(final int limit)}.
52       */
53      private Integer scanLimit = 0;
54  
55      private RangeDeterminingHeuristic<?> rangeStrategy = new RunningMinMaxHeuristic();
56  
57      private TypeDeterminingHeuristic<?> typeStrategy = new StrictHeuristic();
58  
59      private EnumeratedValuesHeuristic<?> enumStrategy = new InMemoryEnumeratedValuesHeuristic();
60

63      /**
64       * The main entry point to application workflow.
65       * 
66       * @param dataUrl Where to find some tabular data.
67       * @return The results of metadata extraction.
68       * @throws IOException
69       */
70      public TabularMetadata getMetadata(final URL dataUrl) throws IOException {
71          // attempt to extract header names
72          // TODO allow a HeaderHeuristic to use more information than the first line of data
73          final List<String> headerNames;
74          try (final CSVParser headerParser = parse(dataUrl, CHARACTER_ENCODING, format)) {
75              final CSVRecord firstLine = headerParser.iterator().next();
76              for (final String field : firstLine) {
77                  headerStrategy.addValue(field);
78              }
79              final boolean hasHeaders = headerStrategy.isHeader();
80              headerStrategy.reset();
81              format = hasHeaders ? format.withHeader() : format;
82              headerNames = hasHeaders ? newArrayList(firstLine) : emptyHeaders(firstLine.size());
83          }
84          // scan values up to the limit
85          final List<TypeDeterminingHeuristic<?>> typeStrategies;
86          final List<RangeDeterminingHeuristic<?>> rangeStrategies;
87          final List<EnumeratedValuesHeuristic<?>> enumStrategies;
88  
89          try (final CSVParser parser = parse(dataUrl, CHARACTER_ENCODING, format)) {
90              final TabularScanner scanner = new TabularScanner(parser, typeStrategy, rangeStrategy, enumStrategy);
91              scanner.scan(scanLimit);
92              typeStrategies = scanner.getTypeStrategies();
93              rangeStrategies = scanner.getRangeStrategies();
94              enumStrategies = scanner.getEnumStrategies();
95          }
96          // extract the results for each field
97          final List<SortedSet<DataType>> columnTypes = transform(typeStrategies, extractType);
98          final List<Map<DataType, Range<?>>> minMaxes = transform(rangeStrategies, extractMinMax);
99          final List<Map<DataType, Set<String>>> enumValues = transform(enumStrategies, extractEnumValues);
100 
101         return new TabularMetadata(headerNames, columnTypes, minMaxes, enumValues);
102     }

104     private static final Function<TypeDeterminingHeuristic<?>, SortedSet<DataType>> extractType =
105             new Function<TypeDeterminingHeuristic<?>, SortedSet<DataType>>() {
106 
107                 @Override
108                 public SortedSet<DataType> apply(final TypeDeterminingHeuristic<?> strategy) {
109                     return strategy.typesAsLikely();
110                 }
111             };
112 
113     private static final Function<RangeDeterminingHeuristic<?>, Map<DataType, Range<?>>> extractMinMax =
114             new Function<RangeDeterminingHeuristic<?>, Map<DataType, Range<?>>>() {
115 
116                 @Override
117                 public Map<DataType, Range<?>> apply(final RangeDeterminingHeuristic<?> strategy) {
118                     return strategy.getRanges();
119                 }
120             };
121 
122     private static final Function<EnumeratedValuesHeuristic<?>, Map<DataType, Set<String>>> extractEnumValues =
123             new Function<EnumeratedValuesHeuristic<?>, Map<DataType, Set<String>>>() {
124 
125                 @Override
126                 public Map<DataType, Set<String>> apply(final EnumeratedValuesHeuristic<?> strategy) {
127                     return strategy.getEnumeratedValues();
128                 }
129             };

131     private static final List<String> emptyHeaders(final int numFields) {
132         final List<String> headers = new ArrayList<>(numFields);
133         for (int i = 0; i <= numFields; i++) {
134             headers.add("Variable " + i);
135         }
136         return headers;
137     }
