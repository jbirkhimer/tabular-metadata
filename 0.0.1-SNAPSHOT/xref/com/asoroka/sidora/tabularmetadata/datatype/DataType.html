45  /**
46   * This enumeration constructs
47   * <ul>
48   * <li>our "native" type system, as well as</li>
49   * <li>maps from lexical spaces in our system to selected portions of selected value spaces in the Java type system,
50   * and</li>
51   * <li>maps from our types to types in the XSD type system.</li>
52   * </ul>
53   * It also offers convenience methods for working with our type system.
54   * 
55   * @author ajs6f
56   */
57  public enum DataType {
58  
59      String(W3C_XML_SCHEMA_NS_URI + "#string", null) {
60  
61          /*
62           * This override indicates that String is the top type.
63           */
64          @Override
65          public EnumSet<DataType> supertypes() {
66              return of((DataType) this);
67          }
68  
69          @SuppressWarnings("unchecked")
70          @Override
71          public String parse(final java.lang.String s) {
72              return s;
73          }
74      },
75      Decimal(W3C_XML_SCHEMA_NS_URI + "#decimal", String) {
76  
77          @SuppressWarnings("unchecked")
78          @Override
79          public Float parse(final java.lang.String s) throws ParsingException {
80              try {
81                  return parseFloat(s.trim());
82              } catch (final NumberFormatException e) {
83                  throw new ParsingException("Could not parse as Decimal!", e);
84              }
85          }
86      },
87      Integer(W3C_XML_SCHEMA_NS_URI + "#integer", Decimal) {
88  
89          @SuppressWarnings("unchecked")
90          @Override
91          public Integer parse(final java.lang.String s) throws ParsingException {
92              try {
93                  return parseInt(s.trim());
94              } catch (final NumberFormatException e) {
95                  throw new ParsingException("Could not parse as Integer!", e);
96              }
97          }
98      },
99      URI(W3C_XML_SCHEMA_NS_URI + "#anyURI", String) {
100 
101         @SuppressWarnings("unchecked")
102         @Override
103         public URI parse(final java.lang.String s) throws ParsingException {
104             try {
105                 if (URI_REGEX.matcher(s).matches()) {
106                     return new URI(s);
107                 }
108                 throw new URISyntaxException(s, "Could not validate URI!");
109             } catch (final URISyntaxException e) {
110                 throw new ParsingException("Could not parse as URI!", e);
111             }
112         }
113     },
114     NonNegativeInteger(W3C_XML_SCHEMA_NS_URI + "#nonNegativeInteger", Integer) {
115 
116         @SuppressWarnings("unchecked")
117         @Override
118         public Integer parse(final java.lang.String s) throws ParsingException {
119 
120             final Integer value = Integer.parse(s);
121             if (value > -1) {
122                 return value;
123             }
124             throw new ParsingException("Value was negative!");
125         }
126     },
127     PositiveInteger(W3C_XML_SCHEMA_NS_URI + "#positiveInteger", NonNegativeInteger) {
128 
129         @SuppressWarnings("unchecked")
130         @Override
131         public Integer parse(final java.lang.String s) throws ParsingException {
132             final Integer value = NonNegativeInteger.parse(s);
133             if (value > 0) {
134                 return value;
135             }
136             throw new ParsingException("Value was negative!");
137         }
138     },
139     Geographic(null, String) {
140 
141         @SuppressWarnings("unchecked")
142         @Override
143         public GeographicValue parse(final String s) throws ParsingException {
144             try {
145                 final List<Float> parts = transform(asList(s.split(",")), string2float);
146                 return new GeographicValue(parts);
147             } catch (final NumberFormatException e) {
148                 throw new ParsingException("Could not parse as Geographic!", e);
149             }
150         }
151     },
152     Boolean(W3C_XML_SCHEMA_NS_URI + "#boolean", String) {
153 
154         @SuppressWarnings("unchecked")
155         @Override
156         public Boolean parse(final java.lang.String s) throws ParsingException {
157             if (BOOLEAN_TRUE_REGEXP.matcher(s).matches()) {
158                 return true;
159             }
160             if (BOOLEAN_FALSE_REGEXP.matcher(s).matches()) {
161                 return false;
162             }
163             throw new ParsingException("Could not parse as Boolean!");
164         }
165     },
166     DateTime(W3C_XML_SCHEMA_NS_URI + "#dateTime", String) {
167 
168         @SuppressWarnings("unchecked")
169         @Override
170         public DateTime parse(final java.lang.String s) throws ParsingException {
171             for (final DateTimeFormatter format : dateTimeFormats) {
172                 try {
173                     return format.parseDateTime(s);
174                 }
175                 catch (final IllegalArgumentException e) {
176                     // log.trace("Could not parse date '{}' in form {}.");
177                 }
178             }
179             throw new ParsingException("Could not parse as DataTime!");
180         }
181     };

183     public static final EnumSet<DataType> valuesSet() {
184         return allOf(DataType.class);
185     }
186 
187     private DataType(final String uri, final DataType supertype) {
188         this.uri = uri == null ? null : java.net.URI.create(uri);
189         this.supertype = supertype;
190     }
191 
192     /**
193      * An URI, if any, with which this DataType is associated
194      */
195     public final URI uri;
196 
197     /**
198      * This DataType, recorded because Java doesn't handle lexical "this" as we might like.
199      */
200     final DataType self = this;
201 
202     /**
203      * The immediate supertype of this DataType.
204      */
205     final DataType supertype;
206 
207     /**
208      * We memoize the calculation of supertypes to avoid redoing this recursion during operation.
209      */
210     private final Supplier<EnumSet<DataType>> supertypesMemo = memoize(new Supplier<EnumSet<DataType>>() {
211 
212         @Override
213         public EnumSet<DataType> get() {
214             final EnumSet<DataType> types = copyOf(supertype.supertypes());
215             types.add(self);
216             return types;
217         }
218     });
219 
220     /**
221      * @return The supertypes of this type, including itself.
222      */
223     public EnumSet<DataType> supertypes() {
224         return supertypesMemo.get();
225     }
226 
227     /**
228      * Map from lex to Java value.
229      * 
230      * @param s
231      * @return s parsed into an instance of a Java type representing this {@link DataType}
232      * @throws ParsingException
233      */
234     abstract public <T extends Comparable<T>> T parse(final String s) throws ParsingException;
235 
236     /**
237      * @param s A value to parse
238      * @return an {@link EnumSet} of those DataTypes into which s can be parsed
239      */
240     public static EnumSet<DataType> parseableAs(final String s) {
241         return copyOf(filter(DataType.valuesSet(), new Predicate<DataType>() {
242 
243             @Override
244             public boolean apply(final DataType t) {
245                 try {
246                     t.parse(s);
247                     return true;
248                 } catch (final ParsingException e) {
249                     return false;
250                 }
251             }
252         }));
253     }
254 
255     private static Function<DataType, Integer> numSuperTypes = new Function<DataType, Integer>() {
256 
257         @Override
258         public Integer apply(final DataType type) {
259             return type.supertypes().size();
260         }
261     };
262 
263     /**
264      * A simple ordering by hierarchy. Those types with more supertypes are considered "smaller" than those with
265      * fewer. In case of a tie, natural enum order prevails.
266      */
267     private static final Comparator<DataType> orderingByHierarchy = natural().reverse()
268             .onResultOf(numSuperTypes).compound(natural());
269 
270     public static SortedSet<DataType> sortByHierarchy(final Iterable<DataType> types) {
271         return from(types).toSortedSet(orderingByHierarchy);
272     }

276     /**
277      * How to recognize a Boolean lex for true.
278      */
279     public static final Pattern BOOLEAN_TRUE_REGEXP = compile("^true|t$", CASE_INSENSITIVE);
280 
281     /**
282      * How to recognize a Boolean lex for false.
283      */
284     public static final Pattern BOOLEAN_FALSE_REGEXP = compile("^false|f$", CASE_INSENSITIVE);
285 
286     /**
287      * We use the well-known regex from <a href="http://tools.ietf.org/html/rfc3986#appendix-B">the standard</a> but
288      * we disallow relative URIs.
289      */
290     static Pattern URI_REGEX = compile("^(([^:/?#]+):)(//([^/?#]*))?([^?#]*)(\\?([^#]*))?(#(.*))?");
291 
292     static final Function<java.lang.String, Float> string2float = new Function<String, Float>() {
293 
294         @Override
295         public Float apply(final java.lang.String seg) {
296             return parseFloat(seg);
297         }
298     };

300     static List<DateTimeFormatter> dateTimeFormats = ImmutableList.of(shortDateTime(), mediumDateTime(),
301             fullDateTime(), dateOptionalTimeParser(), dateTimeParser(), forPattern("yyyy-MM-dd HH:mm:ss"));
