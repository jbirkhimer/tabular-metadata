19  /**
20   * A {@link Heuristic} that aggregates candidate type appearance information for its field.
21   * 
22   * @author ajs6f
23   * @param <SelfType>
24   */
25  public abstract class CountAggregatingHeuristic<SelfType extends CountAggregatingHeuristic<SelfType>> extends
26          ValueCountingHeuristic<SelfType> {
27  
28      /**
29       * In this {@link Map}, we aggregate counts of parseable values for each datatype.
30       */
31      protected EnumMap<DataType, Integer> typeCounts;
32  
33      /**
34       * Initialize counts for each datatype.
35       */
36      @Override
37      public void reset() {
38          super.reset();
39          this.typeCounts = new EnumMap<>(DataType.class);
40          final Map<DataType, Integer> zeroes = toMap(DataType.valuesSet(), constant(0));
41          this.typeCounts.putAll(zeroes);
42      }
43  
44      @Override
45      public void addValue(final String value) {
46          super.addValue(value);
47          incrementCounts(parseableAs(value));
48      }
49  
50      private void incrementCounts(final Collection<DataType> types) {
51          for (final DataType type : types) {
52              typeCounts.put(type, typeCounts.get(type) + 1);
53          }
54      }
55  }
