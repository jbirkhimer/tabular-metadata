23  /**
24   * A {@link DataTypeHeuristic} that aggregates count information for its field.
25   * 
26   * @author ajs6f
27   * @param <T>
28   */
29  public abstract class CountAggregatingHeuristic<T extends CountAggregatingHeuristic<T>> extends
30          RunningMinMaxHeuristic<CountAggregatingHeuristic<T>> {
31  
32      /**
33       * In this {@link Map}, we aggregate counts of parseable values for each datatype.
34       */
35      protected final EnumMap<DataType, Integer> typeCounts = new EnumMap<>(DataType.class);
36  
37      private static final Logger log = getLogger(CountAggregatingHeuristic.class);
38  
39      /**
40       * Initialize counts for each datatype.
41       */
42      public CountAggregatingHeuristic() {
43          for (final DataType type : DataType.values()) {
44              typeCounts.put(type, 0);
45          }
46      }
47  
48      @Override
49      public DataType mostLikelyType() {
50          log.trace("Working with {} total values.", totalNumValues());
51          // we filter to only those types which were selected as parseable for all values
52          final Set<DataType> possibleTypes = filter(typeCounts.keySet(), new Predicate<DataType>() {
53  
54              @Override
55              public boolean apply(final DataType type) {
56                  return candidacy(type);
57              }
58          });
59          log.trace("Found candidate types: {}", possibleTypes);
60          return firstMostRestrictiveType(possibleTypes);
61  
62      }
63  
64      /**
65       * We make use here of the fact that String is the top type. IOW, anything that can be parsed can be parsed as a
66       * String.
67       * 
68       * @return The total number of values seen so far
69       */
70      protected int totalNumValues() {
71          return typeCounts.get(String);
72      }
73  
74      /**
75       * @return Whether this type should be considered as a candidate for selection.
76       */
77      abstract protected boolean candidacy(DataType d);
78  
79      @Override
80      public void addValue(final String value) {
81          super.addValue(value);
82          for (final DataType type : DataType.parseableAs(value)) {
83              typeCounts.put(type, typeCounts.get(type) + 1);
84          }
85      }
86  
87      /*
88       * (non-Javadoc)
89       * @see com.asoroka.sidora.csvmetadata.heuristics.RunningMinMaxHeuristic#clone()
90       */
91      @Override
92      abstract public T clone();
93  
94      @Override
95      public int hashCode() {
96          return hash(typeCounts);
97      }
98  
99      @Override
100     public boolean equals(final Object o) {
101 
102         if (!(CountAggregatingHeuristic.class.isInstance(o))) {
103             return false;
104         }
105         final CountAggregatingHeuristic<?> oo = (CountAggregatingHeuristic<?>) o;
106         return this.hashCode() == (oo.hashCode());
107     }
