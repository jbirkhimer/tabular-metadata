16  import com.asoroka.sidora.tabularmetadata.datatype.DataType;

18  import com.googlecode.totallylazy.Callable1;

21  public class RowsOfRandomValuesForAllTypesSupplier extends ParameterSupplier {
22  
23      @Override
24      public List<PotentialAssignment> getValueSources(final ParameterSignature sig) {
25  
26          final RowsOfRandomValuesForAllTypes metadata = sig.getAnnotation(RowsOfRandomValuesForAllTypes.class);
27          final short valuesPerType = metadata.valuesPerType();
28          final short numRowsPerType = metadata.numRowsPerType();
29  
30          return sequence(DataType.values()).flatMap(rowsOfRandomValuesPerType(numRowsPerType, valuesPerType))
31                  .map(toPotentialAssignment()).toList();
32  
33      }
34  
35      private static Callable1<DataType, Sequence<RandomValuesForAType>> rowsOfRandomValuesPerType(
36              final short numRowsPerType, final short valuesPerType) {
37          return new Callable1<DataType, Sequence<RandomValuesForAType>>() {
38  
39              @Override
40              public Sequence<RandomValuesForAType> call(final DataType type) {
41                  return forwardOnly(range(0, numRowsPerType)).map(randomValuesForAType(valuesPerType, type));
42              }
43          };
44      }
45  
46      static Callable1<Number, RandomValuesForAType> randomValuesForAType(
47              final short valuesPerType, final DataType type) {
48          return new Callable1<Number, RandomValuesForAType>() {
49  
50              @Override
51              public RandomValuesForAType call(final Number input) {
52                  return randomValues(type, valuesPerType);
53              }
54          };
55      }
56  }
