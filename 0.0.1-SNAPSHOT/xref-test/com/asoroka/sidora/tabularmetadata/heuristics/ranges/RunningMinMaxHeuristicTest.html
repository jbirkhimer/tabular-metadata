22   * @author ajs6f

24  @RunWith(Theories.class)
25  public class RunningMinMaxHeuristicTest extends HeuristicTestFrame<RunningMinMaxHeuristic, Map<DataType, Range<?>>> {
26  
27      @DataPoints
28      public static DataType[] datatypes = DataType.values();
29  
30      @Override
31      protected RunningMinMaxHeuristic newTestHeuristic() {
32          return new RunningMinMaxHeuristic();
33      }
34  
35      /**
36       * Asserts that the range of a {@link DataType} for which no values have been seen is {@link Range#all()}.
37       */
38      @Theory
39      public void testMissingLimits(final DataType type) {
40          final RunningMinMaxHeuristic testStrategy = newTestHeuristic();
41          final Range<?> range = testStrategy.results().get(type);
42          assertEquals("Found a defined range where we should not have!", Range.all(), range);
43      }
44  
45      /**
46       * Asserts that for each {@link DataType}, after supplying some comparable values of that type, the range recorded
47       * should be the accurate range of the values supplied. This test does not examine interactions between types
48       * (i.e. for values that could be parsed as several types).
49       */
50      @Theory
51      public void testMinsAndMaxesShouldAlwaysBeFoundPerType(
52              @RowsOfRandomValuesForAllTypes(numRowsPerType = 5, valuesPerType = 5) final RandomValuesForAType values) {
53  
54          final RunningMinMaxHeuristic testStrategy = newTestHeuristic();
55          final Comparable<?> lowest = natural().min(values);
56          final Comparable<?> highest = natural().max(values);
57          addValues(testStrategy, values);
58          final Range<?> range = testStrategy.results().get(values.type);
59          final Comparable<?> calculatedMaxForType = range.upperEndpoint();
60          final Comparable<?> calculatedMinForType = range.lowerEndpoint();
61          assertEquals("Calculated maximum should be the highest value submitted!", highest,
62                  calculatedMaxForType);
63          assertEquals("Calculated minimum should be the lowest value submitted!", lowest,
64                  calculatedMinForType);
65      }
66  }
