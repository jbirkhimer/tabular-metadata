16  import com.asoroka.sidora.tabularmetadata.testframework.RowsOfRandomValuesForAllTypes;
17  import com.asoroka.sidora.tabularmetadata.testframework.TestUtilities.RandomValuesForAType;
18  
19  @RunWith(Theories.class)
20  public class DefaultHeaderHeuristicTest extends HeaderHeuristicTestFrame<DefaultHeaderHeuristic> {
21  
22      @Override
23      protected DefaultHeaderHeuristic newTestHeuristic() {
24          return new DefaultHeaderHeuristic();
25      }
26  
27      @Theory
28      public void testShouldAcceptStringOnlyHeaders(
29              @RowsOfRandomValuesForAllTypes(numRowsPerType = 5, valuesPerType = 5) final RandomValuesForAType values) {
30          assumeThat(values.type, is(String));
31          final DefaultHeaderHeuristic testHeuristic = newTestHeuristic();
32          addValues(testHeuristic, values);
33          assertTrue(testHeuristic.results());
34      }
35  
36      @Theory
37      public void testShouldNotAcceptNonStringHeaders(
38              @RowsOfRandomValuesForAllTypes(numRowsPerType = 5, valuesPerType = 5) final RandomValuesForAType values) {
39          assumeThat(values.type, not(is(String)));
40          final DefaultHeaderHeuristic testHeuristic = newTestHeuristic();
41          addValues(testHeuristic, values);
42          assertFalse(testHeuristic.results());
43      }
44  }
