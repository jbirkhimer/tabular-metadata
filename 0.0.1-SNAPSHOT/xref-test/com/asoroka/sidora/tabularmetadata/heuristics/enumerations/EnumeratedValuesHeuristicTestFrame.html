21  public abstract class EnumeratedValuesHeuristicTestFrame<TestHeuristic extends EnumeratedValuesHeuristic<TestHeuristic>>
22          extends HeuristicTestFrame<TestHeuristic, Map<DataType, Set<String>>> {
23  
24      // TODO clean up sample data and make it more understandable
25  
26      private static Map<DataType, List<String>> simpleData;
27  
28      /**
29       * A map from types, to maps from correct answers, to simple test data.
30       */
31      private static Map<DataType, Map<Set<String>, List<String>>> simpleTestData = new HashMap<>();
32  
33      /**
34       * A map from types, to maps from correct answers, to slightly more realistic test data.
35       */
36      private static Map<DataType, Map<Set<String>, List<String>>> realisticTestData = new HashMap<>();
37  
38      static {
39          simpleData = ImmutableMap.<DataType, List<String>> builder()
40                  .put(DataType.String,
41                          asList("FOO", "FOO", "BAR", "QUUX"))
42                  .put(DataType.Integer,
43                          asList("1", "3", "3", "5"))
44                  .put(DataType.Geographic,
45                          asList("1,2", "1,3", "1,2", "5,6"))
46                  .build();
47          for (final DataType type : simpleData.keySet()) {
48              final List<String> dataList = simpleData.get(type);
49              simpleTestData.put(type, ImmutableMap.<Set<String>, List<String>> builder()
50                      .put(new HashSet<>(dataList), dataList).build());
51          }
52      }
53      static {
54          realisticTestData =
55                  ImmutableMap.<DataType, Map<Set<String>, List<String>>> builder()
56                          .put(DataType.String,
57                                  of((Set<String>) newHashSet("FOO", "BAR", "QUUX"), asList("FOO", "FOO",
58                                          "BAR", "QUUX")))
59                          .put(DataType.Integer,
60                                  of((Set<String>) newHashSet("1", "3", "5"),
61                                          asList("1", "NOTANINTEGER", "3", "3", "6.7", "5")))
62                          .put(DataType.Decimal,
63                                  of((Set<String>) newHashSet("1", "3", "5"),
64                                          asList("1", "NOTADECIMAL", "3", "3", "5")))
65                          .put(DataType.Geographic,
66                                  of((Set<String>) newHashSet("1,2", "1,3", "5,6"),
67                                          asList("1,2", "1,3", "1,2", "5,6", "1", "NOTAGEOGRAPHICVALUE")))
68                          .build();
69      }
70  
71      @Test
72      public void testWithSimpleData() {
73          testWithSampleData(simpleTestData);
74      }
75  
76      @Test
77      public void testWithRealisticData() {
78          testWithSampleData(realisticTestData);
79      }
80  
81      public void testWithSampleData(final Map<DataType, Map<Set<String>, List<String>>> data) {
82          for (final DataType type : data.keySet()) {
83              final Map<Set<String>, List<String>> expectedDataToTestData = data.get(type);
84              for (final Set<String> expectedResults : expectedDataToTestData.keySet()) {
85                  final TestHeuristic testStrategy = newTestHeuristic();
86                  for (final String lex : expectedDataToTestData.get(expectedResults)) {
87                      testStrategy.addValue(lex);
88                  }
89                  final Set<String> results = testStrategy.results().get(type);
90                  assertEquals(expectedResults, results);
91              }
92          }
93      }
