002package com.asoroka.sidora.csvmetadata.test;
003
004import static org.mockito.Mockito.mock;
005import static org.mockito.Mockito.when;
006
007import org.mockito.internal.stubbing.answers.Returns;
008
009import com.asoroka.sidora.csvmetadata.heuristics.DataTypeHeuristic;
010
011/**
012 * Utilities for testing.
013 * 
014 * @author ajs6f
015 */
016public abstract class TestUtilities {
017
018    /**
019     * The following peculiar locution arises from the need to provide "cloneability" while avoiding a recursive mock
020     * 
021     * @return a cloneable mock {@link DataTypeHeuristic}
022     */
023    public static <T extends DataTypeHeuristic<T>> DataTypeHeuristic<T> cloneableMockStrategy(
024            final DataTypeHeuristic<T> strategy) {
025        final DataTypeHeuristic<T> mocked = mock(DataTypeHeuristic.class);
026        final Returns type = new Returns(strategy.mostLikelyType());
027        when(mocked.mostLikelyType()).thenAnswer(type);
028        final Returns range = new Returns(strategy.getRange());
029        when(mocked.getRange()).thenAnswer(range);
030        final Returns cloner = new Returns(strategy);
031        when(mocked.clone()).thenAnswer(cloner);
032        return mocked;
033    }
034
035}
