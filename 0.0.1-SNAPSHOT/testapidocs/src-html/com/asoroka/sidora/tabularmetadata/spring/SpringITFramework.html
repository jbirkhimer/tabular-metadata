023import com.asoroka.sidora.tabularmetadata.TabularMetadata;

028
029/**
030 * Framework for running Spring integration tests.
031 * 
032 * @author ajs6f
033 */
034@RunWith(SpringJUnit4ClassRunner.class)
035public abstract class SpringITFramework {
036
037    protected static final File testDataDir = new File("src/test/resources/test-data");
038
039    protected static URL getTestFile(final String fileName) throws MalformedURLException {
040        return new File(testDataDir, fileName).toURI().toURL();
041    }
042
043    @Inject
044    protected TabularMetadataGenerator testGenerator;
045
046    public TabularMetadata testFile(final URL testFile, final List<DataType> expectedDatatypes,
047            final Range<?> minMaxes, final DataType expectedDataTypeForRangeTest)
048            throws IOException {
049        final TabularMetadata result = testGenerator.getMetadata(testFile);
050        assertEquals("Got incorrect column types!", expectedDatatypes, getFirstElements(result.fieldTypes()));
051        assertEquals("Got wrong range for a field!", minMaxes, result.minMaxes().get(2).get(
052                expectedDataTypeForRangeTest));
053        return result;
054    }
055
056    protected static Range<Integer> getIntRange() {
057        return Range.closed(56, 23423);
058    }
059
060    protected static Range<Float> getFloatRange() {
061        return Range.closed(56F, 23423F);
062    }
063
064    protected static Range<?> getStringRange() {
065        return Range.closed("0056", "SERIAL NUMBER");
066    }
067
068    protected static String testFileSimple = "simple.csv";
069
070    protected static String testFileSlightlyLessSimple = "slightlysimple.csv";
071
072    protected static final List<DataType> SIMPLE_TYPES = asList(String, String, PositiveInteger);
073
074    protected static final List<DataType> SLIGHTLY_SIMPLE_TYPES = asList(String, String, Decimal);
075
076    protected static final List<DataType> STRING_TYPES = asList(String, String, String);
077
078    private static <T> List<T> getFirstElements(final Iterable<SortedSet<T>> inputs) {
079        final ImmutableList.Builder<T> b = builder();
080        for (final SortedSet<T> s : inputs) {
081            b.add(s.first());
082        }
083        return b.build();
084    }
085
086}
