040@RunWith(PowerMockRunner.class)
041@PrepareOnlyThisForTest(CSVParser.class)
042public class TabularMetadataParserTest {
043
044    private static final String testHeaders = "NAME,RANK,SERIAL NUMBER";
045
046    private static final String testRow1 = "\"Kirk\",\"Captain\",0033";
047
048    private static final byte[] testCsv = (testHeaders + "\n" + testRow1).getBytes();
049
050    private static final String MARKER_VALUE = "Picard";
051
052    private static final String testRow2 = "\"" + MARKER_VALUE + "\",\"Captain\",0456";
053
054    private static final byte[] testCsvWithMarker = (testHeaders + "\n" + testRow1 + "\n" + testRow2).getBytes();
055
056    private static final Answer<InputStream> makeStream(final byte[] bytes) {
057        return new Answer<InputStream>() {
058
059            @Override
060            public InputStream answer(final InvocationOnMock invocation) {
061                return new ByteArrayInputStream(copyOf(bytes, bytes.length));
062            }
063        };
064
065    }
066
067    @Mock
068    private DataTypeHeuristic<?> mockSimpleStrategy, mockStrategy;
069
070    @Mock
071    private HeaderHeuristic mockHeaderHeuristic;
072
073    @Mock
074    private DataType mockDataType;
075
076    static <MinMax extends Comparable<?>> Range<MinMax> testRange() {
077        return Range.all();
078    }
079
080    // private static final Logger log = getLogger(TabularMetadataParserTest.class);
081
082    @Before
083    public void setUp() {
084        when(mockSimpleStrategy.mostLikelyType()).thenReturn(mockDataType);
085        final Returns range = new Returns(testRange());
086        when(mockSimpleStrategy.getRange()).thenAnswer(range);
087        mockStrategy = cloneableMockStrategy(mockSimpleStrategy);
088    }
089
090    @Test
091    public void testOperationWithHeaders() throws IOException {
092        when(mockHeaderHeuristic.apply(anyList())).thenReturn(true);
093        final URL mockURL = mock(URL.class);
094
095        when(mockURL.openStream()).thenAnswer(makeStream(testCsv));
096        final TabularMetadataGenerator testParser = new TabularMetadataGenerator();
097        testParser.setStrategy(mockStrategy);
098        testParser.setHeaderStrategy(mockHeaderHeuristic);
099        final TabularMetadata results = testParser.getMetadata(mockURL);
100
101        final List<String> headers = results.headerNames();
102        final List<DataType> types = results.fieldTypes();
103        @SuppressWarnings("rawtypes")
104        // we ignore type-safety here for a simpler unit test
105        final List ranges = results.minMaxes();
106
107        assertEquals(asList(testHeaders.split(",")), headers);
108        assertTrue(all(types, equalTo(mockDataType)));
109        @SuppressWarnings("unchecked")
110        // we ignore type-safety here for a simpler unit test
111        final boolean rangeTest = all(ranges, equalTo(testRange()));
112        assertTrue(rangeTest);
113
114    }
115
116    @Test
117    public void testOperationWithoutHeaders() throws IOException {
118        when(mockHeaderHeuristic.apply(anyList())).thenReturn(false);
119        final URL mockURL = mock(URL.class);
120        when(mockURL.openStream()).thenAnswer(makeStream(testCsv));
121        final TabularMetadataGenerator testParser = new TabularMetadataGenerator();
122        testParser.setStrategy(mockStrategy);
123        testParser.setHeaderStrategy(mockHeaderHeuristic);
124
125        final TabularMetadata results = testParser.getMetadata(mockURL);
126        final List<String> headers = results.headerNames();
127        assertEquals(emptyList(), headers);
128    }
129
130    @Test
131    public <T extends DataTypeHeuristic<T>> void testOperationWithoutHeadersWithScanLimit() throws IOException {
132        when(mockHeaderHeuristic.apply(anyList())).thenReturn(false);
133        final URL mockURL = mock(URL.class);
134        when(mockURL.openStream()).thenAnswer(makeStream(testCsvWithMarker));
135        final MarkingMockDataTypeHeuristic testStrategy = new MarkingMockDataTypeHeuristic(MARKER_VALUE);
136        final TabularMetadataGenerator testParser = new TabularMetadataGenerator();
137        testParser.setStrategy(testStrategy);
138        testParser.setHeaderStrategy(mockHeaderHeuristic);
139        testParser.setScanLimit(2);
140        testParser.getMetadata(mockURL);
141        assertFalse("Discovered a marker in a row we should not have been scanning!", testStrategy.failure);
142        testParser.setScanLimit(3);
143        testParser.getMetadata(mockURL);
144        assertTrue("Failed to discover a marker in a row we should have been scanning!", testStrategy.failure);
145    }
146
147    private static class MarkingMockDataTypeHeuristic implements DataTypeHeuristic<MarkingMockDataTypeHeuristic> {
148
149        /**
150         * @param marker
151         */
152        public MarkingMockDataTypeHeuristic(final String marker) {
153            this.marker = marker;
154        }
155
156        private String marker;
157
158        public boolean failure = false;
159
160        @Override
161        public DataType mostLikelyType() {
162            return Geographic;
163        }
164
165        @Override
166        public MarkingMockDataTypeHeuristic clone() {
167            return this;
168        }
169
170        @Override
171        public void addValue(final String value) {
172            if (value.equals(marker)) {
173                failure = true;
174            }
175
176        }
177
178        @Override
179        public <MinMax extends Comparable<MinMax>> Range<MinMax> getRange() {
180            return testRange();
181        }
182
183        @Override
184        public MarkingMockDataTypeHeuristic get() {
185            return this;
186        }
187    }
