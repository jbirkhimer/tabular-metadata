002package com.asoroka.sidora.tabularmetadata.heuristics.enumerations;
003
004import static com.google.common.collect.ImmutableMap.of;
005import static com.google.common.collect.Sets.newHashSet;
006import static java.util.Arrays.asList;
007import static org.junit.Assert.assertEquals;
008
009import java.util.HashMap;
010import java.util.HashSet;
011import java.util.List;
012import java.util.Map;
013import java.util.Set;
014
015import org.junit.Test;
016
017import com.asoroka.sidora.tabularmetadata.datatype.DataType;
018import com.asoroka.sidora.tabularmetadata.heuristics.HeuristicTestFrame;
019import com.google.common.collect.ImmutableMap;
020
021public abstract class EnumeratedValuesHeuristicTestFrame<TestHeuristic extends EnumeratedValuesHeuristic<TestHeuristic>>
022        extends HeuristicTestFrame<TestHeuristic, Map<DataType, Set<String>>> {
023
024    // TODO clean up sample data and make it more understandable
025
026    private static Map<DataType, List<String>> simpleData;
027
028    /**
029     * A map from types, to maps from correct answers, to simple test data.
030     */
031    private static Map<DataType, Map<Set<String>, List<String>>> simpleTestData = new HashMap<>();
032
033    /**
034     * A map from types, to maps from correct answers, to slightly more realistic test data.
035     */
036    private static Map<DataType, Map<Set<String>, List<String>>> realisticTestData = new HashMap<>();
037
038    static {
039        simpleData = ImmutableMap.<DataType, List<String>> builder()
040                .put(DataType.String,
041                        asList("FOO", "FOO", "BAR", "QUUX"))
042                .put(DataType.Integer,
043                        asList("1", "3", "3", "5"))
044                .put(DataType.Geographic,
045                        asList("1,2", "1,3", "1,2", "5,6"))
046                .build();
047        for (final DataType type : simpleData.keySet()) {
048            final List<String> dataList = simpleData.get(type);
049            simpleTestData.put(type, ImmutableMap.<Set<String>, List<String>> builder()
050                    .put(new HashSet<>(dataList), dataList).build());
051        }
052    }
053    static {
054        realisticTestData =
055                ImmutableMap.<DataType, Map<Set<String>, List<String>>> builder()
056                        .put(DataType.String,
057                                of((Set<String>) newHashSet("FOO", "BAR", "QUUX"), asList("FOO", "FOO",
058                                        "BAR", "QUUX")))
059                        .put(DataType.Integer,
060                                of((Set<String>) newHashSet("1", "3", "5"),
061                                        asList("1", "NOTANINTEGER", "3", "3", "6.7", "5")))
062                        .put(DataType.Decimal,
063                                of((Set<String>) newHashSet("1", "3", "5"),
064                                        asList("1", "NOTADECIMAL", "3", "3", "5")))
065                        .put(DataType.Geographic,
066                                of((Set<String>) newHashSet("1,2", "1,3", "5,6"),
067                                        asList("1,2", "1,3", "1,2", "5,6", "1", "NOTAGEOGRAPHICVALUE")))
068                        .build();
069    }
070
071    @Test
072    public void testWithSimpleData() {
073        testWithSampleData(simpleTestData);
074    }
075
076    @Test
077    public void testWithRealisticData() {
078        testWithSampleData(realisticTestData);
079    }
080
081    public void testWithSampleData(final Map<DataType, Map<Set<String>, List<String>>> data) {
082        for (final DataType type : data.keySet()) {
083            final Map<Set<String>, List<String>> expectedDataToTestData = data.get(type);
084            for (final Set<String> expectedResults : expectedDataToTestData.keySet()) {
085                final TestHeuristic testStrategy = newTestHeuristic();
086                for (final String lex : expectedDataToTestData.get(expectedResults)) {
087                    testStrategy.addValue(lex);
088                }
089                final Set<String> results = testStrategy.results().get(type);
090                assertEquals(expectedResults, results);
091            }
092        }
093    }
094
095}
