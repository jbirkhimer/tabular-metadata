002package com.asoroka.sidora.tabularmetadata.heuristics.types;
003
004import static com.asoroka.sidora.tabularmetadata.datatype.DataType.NonNegativeInteger;
005import static com.asoroka.sidora.tabularmetadata.datatype.DataType.PositiveInteger;
006import static com.asoroka.sidora.tabularmetadata.datatype.DataType.String;
007import static com.asoroka.sidora.tabularmetadata.testframework.TestUtilities.addValues;
008import static java.util.UUID.randomUUID;
009import static org.hamcrest.CoreMatchers.is;
010import static org.hamcrest.core.IsNot.not;
011import static org.junit.Assert.assertEquals;
012import static org.junit.Assert.assertNotEquals;
013import static org.junit.Assert.assertTrue;
014import static org.junit.Assume.assumeThat;
015
016import org.junit.experimental.theories.DataPoints;
017import org.junit.experimental.theories.Theories;
018import org.junit.experimental.theories.Theory;
019import org.junit.runner.RunWith;
020
021import com.asoroka.sidora.tabularmetadata.datatype.DataType;
022import com.asoroka.sidora.tabularmetadata.testframework.RowsOfRandomValuesForAllTypes;
023import com.asoroka.sidora.tabularmetadata.testframework.TestUtilities.RandomValuesForAType;
024
025@RunWith(Theories.class)
026public class FractionHeuristicTest extends PerTypeHeuristicTestFrame<FractionHeuristic> {
027
028    @DataPoints
029    public static DataType[] datatypes = DataType.values();
030
031    @Override
032    protected FractionHeuristic newTestHeuristic() {
033        return new FractionHeuristic(0.2F);
034    }
035
036    @Theory
037    public void inputsWithNoUnparseableValuesShouldBeRecognizedAsTheirTrueType(@RowsOfRandomValuesForAllTypes(
038            numRowsPerType = 5, valuesPerType = 50) final RandomValuesForAType values) {
039        final FractionHeuristic testHeuristic = newTestHeuristic();
040        addValues(testHeuristic, values);
041        final DataType type = values.type;
042        if (type.equals(NonNegativeInteger)) {
043            // NonNegativeInteger and PositiveInteger differ by only one value (0); it's difficult to tell them apart
044            assertTrue(type.equals(NonNegativeInteger) || type.equals(PositiveInteger));
045        } else {
046            assertEquals(type, testHeuristic.mostLikelyType());
047        }
048    }
049
050    @Theory
051    public void testThatInputsWithOnlyOneUnparseableValueShouldBeRecognizedAsTheirTrueType(
052            @RowsOfRandomValuesForAllTypes(
053                    numRowsPerType = 5, valuesPerType = 50) final RandomValuesForAType values) {
054        final FractionHeuristic testHeuristic = newTestHeuristic();
055        // A UUID could only be recognized as a String
056        values.add(randomUUID());
057        addValues(testHeuristic, values);
058        final DataType type = values.type;
059        if (type.equals(NonNegativeInteger)) {
060            // NonNegativeInteger and PositiveInteger differ by only one value (0); it's difficult to tell them apart
061            assertTrue(type.equals(NonNegativeInteger) || type.equals(PositiveInteger));
062        } else {
063            assertEquals(type, testHeuristic.mostLikelyType());
064        }
065    }
066
067    @Theory
068    public void testThatInputsWithHalfUnparseableValuesShouldBeNotRecognizedAsTheirTrueType(
069            @RowsOfRandomValuesForAllTypes(
070                    numRowsPerType = 5, valuesPerType = 50) final RandomValuesForAType values) {
071        // nothing cannot be recognized as a String
072        assumeThat(values.type, not(is(String)));
073
074        final FractionHeuristic testHeuristic = newTestHeuristic();
075        // A UUID could only be recognized as a String
076        final int numValues = values.size();
077        for (byte counter = 0; counter < numValues; counter++) {
078            values.add(randomUUID());
079        }
080        addValues(testHeuristic, values);
081        assertNotEquals(values.type, testHeuristic.mostLikelyType());
082
083    }
084
085    @Theory
086    public void testAcceptAllTypesInTheAbsenceOfData(final DataType type) {
087        final FractionHeuristic testHeuristic = newTestHeuristic();
088        assertTrue(testHeuristic.typesAsLikely().contains(type));
089    }
090}
