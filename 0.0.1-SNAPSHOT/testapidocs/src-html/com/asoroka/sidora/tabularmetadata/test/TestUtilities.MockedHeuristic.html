002package com.asoroka.sidora.tabularmetadata.test;
003
004import static org.mockito.Mockito.mock;
005import static org.mockito.Mockito.when;
006
007import org.mockito.internal.stubbing.answers.Returns;
008
009import com.asoroka.sidora.tabularmetadata.heuristics.enumerations.EnumeratedValuesHeuristic;
010import com.asoroka.sidora.tabularmetadata.heuristics.ranges.RangeDeterminingHeuristic;
011import com.asoroka.sidora.tabularmetadata.heuristics.types.TypeDeterminingHeuristic;
012
013/**
014 * Utilities for testing.
015 * 
016 * @author ajs6f
017 */
018public abstract class TestUtilities {
019
020    /**
021     * The following peculiar locution arises from the need to provide "cloneability" while avoiding a recursive mock
022     * 
023     * @return a cloneable mock strategy
024     */
025    public static MockedHeuristic cloneableMockStrategy(final MockedHeuristic strategy) {
026        final MockedHeuristic mocked = mock(MockedHeuristic.class);
027        final Returns cloner = new Returns(strategy);
028        when(mocked.clone()).thenAnswer(cloner);
029        return mocked;
030    }
031
032    /**
033     * Exists purely to help simplify mocking for tests.
034     * 
035     * @author ajs6f
036     */
037    public static interface MockedHeuristic extends TypeDeterminingHeuristic<MockedHeuristic>,
038            RangeDeterminingHeuristic<MockedHeuristic>,
039            EnumeratedValuesHeuristic<MockedHeuristic> {
040        // NO CONTENT
041    }
042
043}
